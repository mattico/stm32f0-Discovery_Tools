164 gdispImageError gdispImageOpen(gdispImage *img) {
165  gdispImageError err;
166 
167  img->bgcolor = White;
168  for(img->fns = ImageHandlers; img->fns < ImageHandlers+sizeof(ImageHandlers)/sizeof(ImageHandlers[0]); img->fns++) {
169  err = img->fns->open(img);
170  if (err != GDISP_IMAGE_ERR_BADFORMAT) {
171  if ((err & GDISP_IMAGE_ERR_UNRECOVERABLE))
172  img->fns = 0;
173  return err;
174  }
175  img->io.fns->seek(&img->io, 0);
176  }
177  img->type = GDISP_IMAGE_TYPE_UNKNOWN;
178  img->flags = 0;
179  img->fns = 0;
180  img->priv = 0;
181  return GDISP_IMAGE_ERR_BADFORMAT;
182 }
