33 /**
34  * @brief Initialise the mouse/touch.
35  *
36  * @notapi
37  */
38 void ginput_lld_mouse_init(void) {
39  init_board();
40 
41  // Init default values. (From NHD-3.5-320240MF-ATXL-CTP-1 datasheet)
42  // Valid touching detect threshold
43  write_reg(FT5x06_ID_G_THGROUP, 1, 0x16);
44 
45  // valid touching peak detect threshold
46  write_reg(FT5x06_ID_G_THPEAK, 1, 0x3C);
47 
48  // Touch focus threshold
49  write_reg(FT5x06_ID_G_THCAL, 1, 0xE9);
50 
51  // threshold when there is surface water
52  write_reg(FT5x06_ID_G_THWATER, 1, 0x01);
53 
54  // threshold of temperature compensation
55  write_reg(FT5x06_ID_G_THTEMP, 1, 0x01);
56 
57  // Touch difference threshold
58  write_reg(FT5x06_ID_G_THDIFF, 1, 0xA0);
59 
60  // Delay to enter 'Monitor' status (s)
61  write_reg(FT5x06_ID_G_TIME_ENTER_MONITOR, 1, 0x0A);
62 
63  // Period of 'Active' status (ms)
64  write_reg(FT5x06_ID_G_PERIODACTIVE, 1, 0x06);
65 
66  // Timer to enter ‘idle’ when in 'Monitor' (ms)
67  write_reg(FT5x06_ID_G_PERIODMONITOR, 1, 0x28);
68 }

70 /**
71  * @brief Read the mouse/touch position.
72  *
73  * @param[in] pt A pointer to the structure to fill
74  *
75  * @note For drivers that don't support returning a position
76  * when the touch is up (most touch devices), it should
77  * return the previous position with the new Z value.
78  * The z value is the pressure for those touch devices
79  * that support it (-100 to 100 where > 0 is touched)
80  * or, 0 or 100 for those drivers that don't.
81  *
82  * @notapi
83  */
84 void ginput_lld_mouse_get_reading(MouseReading *pt) {
85  // Poll to get the touched status
86  uint8_t last_touched;
87 
88  last_touched = touched;
89  touched = (uint8_t)read_reg(FT5x06_TOUCH_POINTS, 1) & 0x07;
90 
91  // If not touched, return the previous results
92  if (touched == 0) {
93  pt->x = x;
94  pt->y = y;
95  pt->z = 0;
96  pt->buttons = 0;
97  return;
98  }
99 
100  /* Get the X, Y, Z values */
101  x = (coord_t)(read_reg(FT5x06_TOUCH1_XH, 2) & 0x0fff);
102  y = (coord_t)read_reg(FT5x06_TOUCH1_YH, 2);
103  z = 100;
104 
105  // Rescale X,Y,Z - X & Y don't need scaling when you are using calibration!
106 #if !GINPUT_MOUSE_NEED_CALIBRATION
107  x = gdispGetWidth() - x / (4096/gdispGetWidth());
108  y = y / (4096/gdispGetHeight());
109 #endif
110 
111  // Return the results. ADC gives values from 0 to 2^12 (4096)
112  pt->x = x;
113  pt->y = y;
114  pt->z = z;
115  pt->buttons = GINPUT_TOUCH_PRESSED;
116 }
117 
118 #endif /* GFX_USE_GINPUT && GINPUT_NEED_MOUSE */
