107 static bool_t WM_Add(GHandle gh, const GWindowInit *pInit) {
108  // Note the window will not be marked as visible yet

118 static void WM_Delete(GHandle gh) {
119  // Make the window invisible and clear the area underneath
120  if ((gh->flags & GWIN_FLG_VISIBLE)) {

143 static void WM_Redim(GHandle gh, coord_t x, coord_t y, coord_t w, coord_t h) {
144  // This is the simplest way of doing it - just clip the the screen
145  // If it won't fit on the screen move it around until it does.
146  if (x < 0) { w += x; x = 0; }
147  if (y < 0) { h += y; y = 0; }
148  if (x > gdispGetWidth()-MIN_WIN_WIDTH) x = gdispGetWidth()-MIN_WIN_WIDTH;
149  if (y > gdispGetHeight()-MIN_WIN_HEIGHT) y = gdispGetHeight()-MIN_WIN_HEIGHT;
150  if (w < MIN_WIN_WIDTH) { w = MIN_WIN_WIDTH; }
151  if (h < MIN_WIN_HEIGHT) { h = MIN_WIN_HEIGHT; }
152  if (x+w > gdispGetWidth()) w = gdispGetWidth() - x;
153  if (y+h > gdispGetHeight()) h = gdispGetHeight() - y;
154 
155  // If there has been no resize just exit
156  if (gh->x == x && gh->y == y && gh->width == w && gh->height == h)
157  return;
158 
159  // Clear the old area
160  if ((gh->flags & GWIN_FLG_VISIBLE))
161  gdispFillArea(gh->x, gh->y, gh->width, gh->height, gwinGetDefaultBgColor());
162 
163  // Set the new size
164  gh->x = x; gh->y = y;
165  gh->width = w; gh->height = h;
166 
167  // Redraw the window (if possible)
168  if ((gh->flags & GWIN_FLG_VISIBLE)) {
169  if (gh->vmt->Redraw) {
170  #if GDISP_NEED_CLIP
171  gdispSetClip(gh->x, gh->y, gh->width, gh->height);
172  #endif
173  gh->vmt->Redraw(gh);
174  }
175  }
176 }

183 static void WM_Raise(GHandle gh) {
184  // Take it off the list and then put it back on top
185  // The order of the list then reflects the z-order.
186  gfxQueueASyncRemove(&_GWINList, &gh->wmq);
187  gfxQueueASyncPut(&_GWINList, &gh->wmq);
188 
189  // Redraw the window
190  if ((gh->flags & GWIN_FLG_VISIBLE)) {
191  if (gh->vmt->Redraw) {
192  #if GDISP_NEED_CLIP
193  gdispSetClip(gh->x, gh->y, gh->width, gh->height);
194  #endif
195  gh->vmt->Redraw(gh);
196  }
197  }
