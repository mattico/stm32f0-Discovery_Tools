34  static void AudGTimerCallback(void *param) {
35  (void) param;
36  GSourceListener *psl;
37  GEventADC *pe;
38 
39  psl = 0;
40  while ((psl = geventGetSourceListener((GSourceHandle)(&aud), psl))) {
41  if (!(pe = (GEventAudioIn *)geventGetEventBuffer(psl))) {
42  // This listener is missing - save this.
43  psl->srcflags |= GADC_AUDIO_IN_LOSTEVENT;
44  continue;
45  }
46 
47  pe->type = GEVENT_AUDIO_IN;
48  pe->channel = aud.channel;
49  pe->count = lastcount;
50  pe->buffer = lastbuffer;
51  pe->flags = psl->srcflags;
52  psl->srcflags = 0;
53  geventSendEvent(psl);
54  }
55  }

95 bool_t gaudinInit(uint16_t channel, uint32_t frequency, audin_sample_t *buffer, size_t bufcount, size_t samplesPerEvent) {
96  /* Check the channel is valid */
97  if (channel >= GAUDIN_NUM_CHANNELS || frequency > GAUDIN_MAX_SAMPLE_FREQUENCY)
98  return FALSE;
99 
100  /* Stop any existing transfers */
101  if ((audFlags & AUDFLG_RUNNING))
102  gadc_lld_stop();
103  audFlags = 0;
104 
105  /* Initialise everything */
106  aud.channel = channel;
107  aud.frequency = frequency;
108  aud.buffer = buffer;
109  aud.bufcount = bufcount;
110  aud.samplesPerEvent = samplesPerEvent;
111  paudSem = 0;
112  paudEvent = 0;
113 
114  /* Set up the low level driver */
115  gaudin_lld_init(&aud);
116  return TRUE;
117 }
