164  /**
165  * @brief Open an image ready for drawing
166  * @details Determine the image format and get ready to decode the first image frame
167  * @return GDISP_IMAGE_ERR_OK (0) on success or an error code.
168  *
169  * @param[in] img The image structure
170  *
171  * @pre The io fields should be filled in before calling gdispImageOpen()
172  *
173  * @note This determines which decoder to use and then initialises all other fields
174  * in the gdispImage structure.
175  * @note The image background color is set to White.
176  * @note There are three types of return - everything OK, partial success and unrecoverable
177  * failures. For everything OK it returns GDISP_IMAGE_ERR_OK. A partial success can
178  * be distinguished from a unrecoverable failure by testing the GDISP_IMAGE_ERR_UNRECOVERABLE
179  * bit in the error code.
180  * A partial success return code means an image can still be drawn but perhaps with
181  * reduced functionality eg only the first page of a multi-page image.
182  * @note @p gdispImageClose() can be called even after a failure to open the image to ensure
183  * that the IO close routine gets called.
184  */
185  gdispImageError gdispImageOpen(gdispImage *img);

224  /**

227  * @return GDISP_IMAGE_ERR_OK (0) on success or an error code.
228  *
229  * @param[in] img The image structure
230  *
231  * @pre gdispImageOpen() must have returned successfully.
232  *
233  * @note This can use a LOT of RAM!
234  * @note The decoder may choose to ignore the request for caching. If it does so it will

236  * @note A fatal error here does not necessarily mean that drawing the image will fail. For

242  /**
243  * @brief Draw the image
244  * @return GDISP_IMAGE_ERR_OK (0) on success or an error code.
245  *
246  * @param[in] img The image structure
247  * @param[in] x,y The screen location to draw the image
248  * @param[in] cx,cy The area on the screen to draw
249  * @param[in] sx,sy The image position to start drawing at
250  *

254  * is simply not drawn.
255  * @note If @p gdispImageCache() has been called first for this frame, this routine will draw using a

258  * uses a lot less RAM)

262  /**
263  * @brief Prepare for the next frame/page in the image file.
264  * @return A time in milliseconds to keep displaying the current frame before trying to draw
265  * the next frame. Watch out for the special values TIME_IMMEDIATE and TIME_INFINITE.
266  *
267  * @param[in] img The image structure
268  *
269  * @pre gdispImageOpen() must have returned successfully.
270  *
271  * @note It will return TIME_IMMEDIATE if the first frame/page hasn't been drawn or if the next frame
272  * should be drawn immediately.
273  * @note It will return TIME_INFINITE if another image frame doesn't exist or an error has occurred.
274  * @note Images that support multiple pages (eg TIFF files) will return TIME_IMMEDIATE between pages
275  * and then TIME_INFINITE when there are no more pages.
276  * @note An image that displays a looped animation will never return TIME_INFINITE unless it
277  * gets an error.
278  * @note Calling gdispImageDraw() after getting a TIME_INFINITE will go back to drawing the first
279  * frame/page.
280  */
281  delaytime_t gdispImageNext(gdispImage *img);
