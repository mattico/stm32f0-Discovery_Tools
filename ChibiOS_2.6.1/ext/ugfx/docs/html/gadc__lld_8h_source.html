28 /**
29  * @brief The structure passed to start a timer conversion
30  * @note We use the structure instead of parameters purely to save
31  * interrupt stack space which is very limited in some platforms.
32  * @{
33  */
34 typedef struct GadcLldTimerData_t {
35  uint32_t physdev; /* @< A value passed to describe which physical ADC devices/channels to use. */
36  adcsample_t *buffer; /* @< The static buffer to put the ADC samples into. */
37  size_t count; /* @< The number of conversions to do before doing a callback and stopping the ADC. */
38  bool_t now; /* @< Trigger the first conversion now rather than waiting for the first timer interrupt (if possible) */

115 /**
116  * @brief Start a periodic timer for high frequency conversions.
117  *
118  * @param[in] physdev A value passed to describe which physical ADC devices/channels to use.
119  * @param[in] frequency The frequency to create ADC conversions
120  *
121  * @note The exact meaning of physdev is hardware dependent. It describes the channels
122  * the will be used later on when a "timer" conversion is actually scheduled.
123  * @note It is assumed that the timer is capable of free-running even when the ADC
124  * is stopped or doing something else.
125  * @details When a timer interrupt occurs a conversion should start if these is a "timer" conversion
126  * active.
127  * @note If the ADC is stopped, doesn't have a "timer" conversion active or is currently executing
128  * a non-timer conversion then the interrupt can be ignored other than (optionally) incrementing
129  * the GADC_Timer_Missed variable.
130  *
131  * @api
132  */
133 void gadc_lld_start_timer(uint32_t physdev, uint32_t frequency);

147 /**
148  * @brief Start a "timer" conversion.
149  * @details Starts a series of conversions triggered by the timer.
150  *
151  * @param[in] pgtd Contains the parameters for the timer conversion.
152  *
153  * @note The exact meaning of physdev is hardware dependent. It is likely described in the
154  * drivers gadc_lld_config.h
155  * @note Some versions of ChibiOS actually call the callback function more than once, once
156  * at the half-way point and once on completion. The high level code handles this.
157  * @note The driver should call @p GADC_ISR_CompleteI() when it completes the operation
158  * (or at the half-way point), or @p GAD_ISR_ErrorI() on an error.
159  * @note The high level code ensures that this is not called while a non-timer conversion is in
160  * progress

162  * @iclass
