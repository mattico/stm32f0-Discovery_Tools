36 gdispImageError gdispImageOpen_NATIVE(gdispImage *img) {
37  uint8_t hdr[HEADER_SIZE];
38 
39  /* Read the 8 byte header */
40  if (img->io.fns->read(&img->io, hdr, 8) != 8)
41  return GDISP_IMAGE_ERR_BADFORMAT; // It can't be us
42 
43  if (hdr[0] != 'N' || hdr[1] != 'I')
44  return GDISP_IMAGE_ERR_BADFORMAT; // It can't be us
45 
46  if (hdr[6] != GDISP_PIXELFORMAT/256 || hdr[7] != (GDISP_PIXELFORMAT & 0xFF))
47  return GDISP_IMAGE_ERR_UNSUPPORTED; // Unsupported pixel format
48 
49  /* We know we are a native format image */
50  img->flags = 0;
51  img->width = (((uint16_t)hdr[2])<<8) | (hdr[3]);
52  img->height = (((uint16_t)hdr[4])<<8) | (hdr[5]);
53  if (img->width < 1 || img->height < 1)
54  return GDISP_IMAGE_ERR_BADDATA;
55  if (!(img->priv = (gdispImagePrivate *)gdispImageAlloc(img, sizeof(gdispImagePrivate))))
56  return GDISP_IMAGE_ERR_NOMEMORY;
57  img->priv->frame0cache = 0;
58 
59  img->type = GDISP_IMAGE_TYPE_NATIVE;
60  return GDISP_IMAGE_ERR_OK;
61 }

73 gdispImageError gdispImageCache_NATIVE(gdispImage *img) {
74  size_t len;
75 
76  /* If we are already cached - just return OK */
77  if (img->priv->frame0cache)
78  return GDISP_IMAGE_ERR_OK;
79 
80  /* We need to allocate the cache */
81  len = img->width * img->height * sizeof(pixel_t);
82  img->priv->frame0cache = (pixel_t *)gdispImageAlloc(img, len);
83  if (!img->priv->frame0cache)
84  return GDISP_IMAGE_ERR_NOMEMORY;
85 
86  /* Read the entire bitmap into cache */
87  img->io.fns->seek(&img->io, FRAME0POS);
88  if (img->io.fns->read(&img->io, img->priv->frame0cache, len) != len)
89  return GDISP_IMAGE_ERR_BADDATA;
90 
91  return GDISP_IMAGE_ERR_OK;
92 }

94 gdispImageError gdispImageDraw_NATIVE(gdispImage *img, coord_t x, coord_t y, coord_t cx, coord_t cy, coord_t sx, coord_t sy) {
95  coord_t mx, mcx;
96  size_t pos, len;
97 
98  /* Check some reasonableness */
99  if (sx >= img->width || sy >= img->height) return GDISP_IMAGE_ERR_OK;
100  if (sx + cx > img->width) cx = img->width - sx;
101  if (sy + cy > img->height) cy = img->height - sy;
102 
103  /* Draw from the image cache - if it exists */
104  if (img->priv->frame0cache) {
105  gdispBlitAreaEx(x, y, cx, cy, sx, sy, img->width, img->priv->frame0cache);
106  return GDISP_IMAGE_ERR_OK;
107  }
108 
109  /* For this image decoder we cheat and just seek straight to the region we want to display */
110  pos = FRAME0POS + (img->width * sy + cx) * sizeof(pixel_t);
111 
112  /* Cycle through the lines */
113  for(;cy;cy--, y++) {
114  /* Move to the start of the line */
115  img->io.fns->seek(&img->io, pos);
116 
117  /* Draw the line in chunks using BitBlt */
118  for(mx = x, mcx = cx; mcx > 0; mcx -= len, mx += len) {
119  // Read the data
120  len = img->io.fns->read(&img->io,
121  img->priv->buf,
122  mx > BLIT_BUFFER_SIZE ? (BLIT_BUFFER_SIZE*sizeof(pixel_t)) : (mx * sizeof(pixel_t)))
123  / sizeof(pixel_t);
124  if (!len)
125  return GDISP_IMAGE_ERR_BADDATA;
126 
127  /* Blit the chunk of data */
128  gdispBlitAreaEx(mx, y, len, 1, 0, 0, len, img->priv->buf);
129  }
130 
131  /* Get the position for the start of the next line */
132  pos += img->width*sizeof(pixel_t);
133  }
134 
135  return GDISP_IMAGE_ERR_OK;
