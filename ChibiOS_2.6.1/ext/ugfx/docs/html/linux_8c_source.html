76 gfxThreadHandle gfxThreadCreate(void *stackarea, size_t stacksz, threadpriority_t prio, DECLARE_THREAD_FUNCTION((*fn),p), void *param) {
77  gfxThreadHandle th;
78 
79  // Implementing priority with pthreads is a rats nest that is also pthreads implementation dependent.
80  // Only some pthreads schedulers support it, some implementations use the operating system process priority mechanisms.
81  // Even those that do support it can have different ranges of priority and "normal" priority is an undefined concept.
82  // Across different UNIX style operating systems things can be very different (let alone OS's such as Windows).
83  // Even just Linux changes the way priority works with different kernel schedulers and across kernel versions.
84  // For these reasons we ignore the priority.
85 
86  if (pthread_create(&th, 0, fn, param))
87  return 0;
88  return th;
89 }

113 bool_t gfxSemWait(gfxSem *pSem, delaytime_t ms) {
114  pthread_mutex_lock(&pSem->mtx);
115  switch (ms) {
116  case TIME_INFINITE:
117  while (!pSem->cnt)
118  pthread_cond_wait(&pSem->cond, &pSem->mtx);
119  break;
120  case TIME_IMMEDIATE:
121  if (!pSem->cnt) {
122  pthread_mutex_unlock(&pSem->mtx);
123  return FALSE;
124  }
125  break;
126  default:
127  {
128  struct timeval now;
129  struct timespec tm;
130 
131  gettimeofday(&now);
132  tm.tv_sec = now.tv_sec + ms / 1000;
133  tm.tv_nsec = (now.tv_usec + ms % 1000) * 1000;
134  while (!pSem->cnt) {
135  if (pthread_cond_timedwait(&pSem->cond, &pSem->mtx, &tm) == ETIMEDOUT) {
136  pthread_mutex_unlock(&pSem->mtx);
137  return FALSE;
138  }
139  }
140  }
141  break;
142  }
143  pSem->cnt--;
144  pthread_mutex_unlock(&pSem->mtx);
145  return TRUE;
146 }

148 void gfxSemSignal(gfxSem *pSem) {
149  pthread_mutex_lock(&pSem->mtx);
150  if (pSem->cnt < pSem->max) {
151  pSem->cnt++;
152  pthread_cond_signal(&pSem->cond);
153  }
154  pthread_mutex_unlock(&pSem->mtx);
155 }

157 semcount_t gfxSemCounter(gfxSem *pSem) {
158  semcount_t res;
159 
160  // The locking is really only required if obtaining the count is a divisible operation
161  // which it might be on a 8/16 bit processor with a 32 bit semaphore count.
162  pthread_mutex_lock(&pSem->mtx);
163  res = pSem->cnt;
164  pthread_mutex_unlock(&pSem->mtx);
165  return res;
