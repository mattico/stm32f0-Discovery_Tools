50  static void _gwm_redim(GHandle gh, coord_t x, coord_t y, coord_t width, coord_t height) {
51  gh->x = x; gh->y = y;
52  gh->width = width; gh->height = height;
53  if (gh->x < 0) { gh->width += gh->x; gh->x = 0; }
54  if (gh->y < 0) { gh->height += gh->y; gh->y = 0; }
55  if (gh->x > gdispGetWidth()-MIN_WIN_WIDTH) gh->x = gdispGetWidth()-MIN_WIN_WIDTH;
56  if (gh->y > gdispGetHeight()-MIN_WIN_HEIGHT) gh->y = gdispGetHeight()-MIN_WIN_HEIGHT;
57  if (gh->width < MIN_WIN_WIDTH) { gh->width = MIN_WIN_WIDTH; }
58  if (gh->height < MIN_WIN_HEIGHT) { gh->height = MIN_WIN_HEIGHT; }
59  if (gh->x+gh->width > gdispGetWidth()) gh->width = gdispGetWidth() - gh->x;
60  if (gh->y+gh->height > gdispGetHeight()) gh->height = gdispGetHeight() - gh->y;
61 
62  // Redraw the window
63  if ((gh->flags & GWIN_FLG_VISIBLE)) {
64  if (gh->vmt->Redraw) {
65  #if GDISP_NEED_CLIP
66  gdispSetClip(gh->x, gh->y, gh->width, gh->height);
67  #endif
68  gh->vmt->Redraw(gh);
69  }
70  }
71  }
72 #endif

91 // Internal routine for use by GWIN components only
92 // Initialise a window creating it dynamically if required.
93 GHandle _gwindowCreate(GWindowObject *pgw, const GWindowInit *pInit, const gwinVMT *vmt, uint16_t flags) {
94  // Allocate the structure if necessary
95  if (!pgw) {
96  if (!(pgw = (GWindowObject *)gfxAlloc(vmt->size)))
97  return 0;
98  pgw->flags = flags|GWIN_FLG_DYNAMIC;
99  } else
100  pgw->flags = flags;
101 
102  // Initialise all basic fields
103  pgw->vmt = vmt;
104  pgw->color = defaultFgColor;
105  pgw->bgcolor = defaultBgColor;
106  #if GDISP_NEED_TEXT
107  pgw->font = defaultFont;
108  #endif
109 
110 #if GWIN_NEED_WINDOWMANAGER
111  if (!_GWINwm->vmt->Add(pgw, pInit)) {
112  if ((pgw->flags & GWIN_FLG_DYNAMIC))
113  gfxFree(pgw);
114  return 0;
115  }
116 #else
117  _gwm_redim(pgw, pInit->x, pInit->y, pInit->width, pInit->height);
118 #endif
119 
120  return (GHandle)pgw;
121 }

187  if (visible) {

189  gh->flags |= GWIN_FLG_VISIBLE;

198  gh->flags &= ~GWIN_FLG_VISIBLE;
199  #if GWIN_NEED_WINDOWMANAGER
