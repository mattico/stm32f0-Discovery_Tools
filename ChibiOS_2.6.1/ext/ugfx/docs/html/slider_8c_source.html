32 // Send the slider event
33 static void SendSliderEvent(GWidgetObject *gw) {
34  GSourceListener * psl;
35  GEvent * pe;
36  #define pse ((GEventGWinSlider *)pe)
37 
38  // Trigger a GWIN Button Event
39  psl = 0;
40  while ((psl = geventGetSourceListener(GWIDGET_SOURCE, psl))) {
41  if (!(pe = geventGetEventBuffer(psl)))
42  continue;
43  pse->type = GEVENT_GWIN_SLIDER;
44  pse->slider = (GHandle)gw;
45  pse->position = ((GSliderObject *)gw)->pos;
46  geventSendEvent(psl);
47  }
48 
49  #undef pse
50 }

52 // Reset the display position back to the value predicted by the saved slider position
53 static void ResetDisplayPos(GSliderObject *gsw) {
54  if (gsw->w.g.width < gsw->w.g.height)
55  gsw->dpos = gsw->w.g.height-1-((gsw->w.g.height-1)*(gsw->pos-gsw->min))/(gsw->max-gsw->min);
56  else
57  gsw->dpos = ((gsw->w.g.width-1)*(gsw->pos-gsw->min))/(gsw->max-gsw->min);
58 }

60 #if GINPUT_NEED_MOUSE
61  // A mouse up event
62  static void MouseUp(GWidgetObject *gw, coord_t x, coord_t y) {
63  #define gsw ((GSliderObject *)gw)
64  #define gh ((GHandle)gw)
65 
66  #if GWIN_BUTTON_LAZY_RELEASE
67  // Clip to the slider
68  if (x < 0) x = 0;
69  else if (x >= gh->width) x = gh->width-1;
70  if (y < 0) y = 0;
71  else if (y >= gh->height) x = gh->height-1;
72  #else
73  // Are we over the slider?
74  if (x < 0 || x >= gh->width || y < 0 || y >= gh->height) {
75  // No - restore the slider
76  ResetDisplayPos(gsw);
77  _gwidgetRedraw(gh);
78  return;
79  }
80  #endif
81 
82  // Set the new position
83  if (gh->width < gh->height) {
84  if (y > gh->height-GWIN_SLIDER_DEAD_BAND)
85  gsw->pos = gsw->min;
86  else if (y < GWIN_SLIDER_DEAD_BAND)
87  gsw->pos = gsw->max;
88  else
89  gsw->pos = (uint16_t)((int32_t)(gh->height-1-y-GWIN_SLIDER_DEAD_BAND)*(gsw->max-gsw->min)/(gh->height-2*GWIN_SLIDER_DEAD_BAND) + gsw->min);
90  } else {
91  if (x > gh->width-GWIN_SLIDER_DEAD_BAND)
92  gsw->pos = gsw->max;
93  else if (x < GWIN_SLIDER_DEAD_BAND)
94  gsw->pos = gsw->min;
95  else
96  gsw->pos = (uint16_t)((int32_t)(x-GWIN_SLIDER_DEAD_BAND)*(gsw->max-gsw->min)/(gh->width-2*GWIN_SLIDER_DEAD_BAND) + gsw->min);
97  }
98 
99  ResetDisplayPos(gsw);
100  _gwidgetRedraw(gh);
101 
102  // Generate the event
103  SendSliderEvent(gw);
104  #undef gh
105  #undef gsw
106  }

109  static void MouseMove(GWidgetObject *gw, coord_t x, coord_t y) {
110  #define gsw ((GSliderObject *)gw)
111 
112  // Determine the temporary display position (with range checking)
113  if (gw->g.width < gw->g.height) {
114  if (y < 0)
115  gsw->dpos = 0;
116  else if (y >= gw->g.height)
117  gsw->dpos = gw->g.height-1;
118  else
119  gsw->dpos = y;
120  } else {
121  if (x < 0)
122  gsw->dpos = 0;
123  else if (x >= gw->g.width)
124  gsw->dpos = gw->g.width-1;
125  else
126  gsw->dpos = x;
127  }
128 
129  // Update the display
130  _gwidgetRedraw(&gw->g);
131  #undef gsw
132  }

244 void gwinSliderSetRange(GHandle gh, int min, int max) {
245  #define gsw ((GSliderObject *)gh)
246 
247  if (gh->vmt != (gwinVMT *)&sliderVMT)
248  return;
249 
250  if (min == max) // prevent divide by 0 errors.
251  max++;
252  gsw->min = min;
253  gsw->max = max;
254  gsw->pos = min;
255  ResetDisplayPos(gsw);
256  #undef gsw

259 void gwinSliderSetPosition(GHandle gh, int pos) {
260  #define gsw ((GSliderObject *)gh)
261 
262  if (gh->vmt != (gwinVMT *)&sliderVMT)
263  return;
264 
265  if (gsw->min <= gsw->max) {
266  if (pos < gsw->min) gsw->pos = gsw->min;
267  else if (pos > gsw->max) gsw->pos = gsw->max;
268  else gsw->pos = pos;
269  } else {
270  if (pos > gsw->min) gsw->pos = gsw->min;
271  else if (pos < gsw->max) gsw->pos = gsw->max;
272  else gsw->pos = pos;
273  }
274  ResetDisplayPos(gsw);
275  #undef gsw
276 }
