253 /**
254  * @brief Set the cursor shape.
255  *
256  * @param[in] shape The shape to set the cursor.
257  * Valid shapes are:
258  * cursorOff = hide cursor
259  * cursorBlock = show cursor as a block
260  * cursorBlinkingBlock = show cursor as a blinking block
261  * cursorUnderline = show a cursor as a thin line under a character
262  * cursorBlinkingUnderline = show a cursor as a blinking thin line under a character
263  * cursorBar = show a cursor as a thin vertical bar near a character
264  * cursorBlinkingBar = show a cursor as a blinking thin vertical bar near a character
265  * @note Not all shapes are necessarily supported. The driver will make a similar
266  * choice if the one specified is not available.
267  */
268 #define tdispSetCursorShape(shape) tdispControl(TDISP_CTRL_CURSOR, (cursorshape)(shape))
