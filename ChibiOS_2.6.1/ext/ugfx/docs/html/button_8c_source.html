32 // Send the button event
33 static void SendButtonEvent(GWidgetObject *gw) {
34  GSourceListener * psl;
35  GEvent * pe;
36  #define pbe ((GEventGWinButton *)pe)
37 
38  // Trigger a GWIN Button Event
39  psl = 0;
40  while ((psl = geventGetSourceListener(GWIDGET_SOURCE, psl))) {
41  if (!(pe = geventGetEventBuffer(psl)))
42  continue;
43  pbe->type = GEVENT_GWIN_BUTTON;
44  pbe->button = (GHandle)gw;
45  geventSendEvent(psl);
46  }
47 
48  #undef pbe
49 }

54  (void) x; (void) y;

59  // A mouse up has occurred (it may or may not be over the button)
60  static void MouseUp(GWidgetObject *gw, coord_t x, coord_t y) {
61  (void) x; (void) y;
62  gw->g.flags &= ~GBUTTON_FLG_PRESSED;
63  _gwidgetRedraw((GHandle)gw);
64 
65  #if !GWIN_BUTTON_LAZY_RELEASE
66  // If the mouse up was not over the button then cancel the event
67  if (x < 0 || y < 0 || x >= gw->g.width || y >= gw->g.height)
68  return;
69  #endif
70 
71  SendButtonEvent(gw);
72  }
73 #endif
