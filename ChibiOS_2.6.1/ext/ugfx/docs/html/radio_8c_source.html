27 // Send the button event
28 static void SendRadioEvent(GWidgetObject *gw) {
29  GSourceListener * psl;
30  GEvent * pe;
31  #define pbe ((GEventGWinRadio *)pe)
32 
33  // Trigger a GWIN Button Event
34  psl = 0;
35  while ((psl = geventGetSourceListener(GWIDGET_SOURCE, psl))) {
36  if (!(pe = geventGetEventBuffer(psl)))
37  continue;
38  pbe->type = GEVENT_GWIN_RADIO;
39  pbe->radio = (GHandle)gw;
40  pbe->group = ((GRadioObject *)gw)->group;
41  geventSendEvent(psl);
42  }
43 
44  #undef pbe
45 }

123 void gwinRadioPress(GHandle gh) {
124  GHandle gx;
125 
126  if (gh->vmt != (gwinVMT *)&radioVMT || (gh->flags & GRADIO_FLG_PRESSED))
127  return;
128 
129  if ((gx = gwinRadioGetActive(((GRadioObject *)gh)->group))) {
130  gx->flags &= ~GRADIO_FLG_PRESSED;
131  _gwidgetRedraw(gx);
132  }
133  gh->flags |= GRADIO_FLG_PRESSED;
134  _gwidgetRedraw(gh);
135  SendRadioEvent((GWidgetObject *)gh);
136 }

145 GHandle gwinRadioGetActive(uint16_t group) {
146  const gfxQueueASyncItem * qi;
147  GHandle gh;
148 
149  for(qi = gfxQueueASyncPeek(&_GWINList); qi; qi = gfxQueueASyncNext(qi)) {
150  gh = QItem2GWindow(qi);
151  if (gh->vmt == (gwinVMT *)&radioVMT && ((GRadioObject *)gh)->group == group && (gh->flags & GRADIO_FLG_PRESSED))
152  return gh;
153  }
154  return 0;
155 }

167 void gwinRadioDraw_Radio(GWidgetObject *gw, void *param) {
168  #define gcw ((GRadioObject *)gw)
169  coord_t ld, df;
170  const GColorSet * pcol;
171  (void) param;
172 
173  if (gw->g.vmt != (gwinVMT *)&radioVMT) return;
174  pcol = getDrawColors(gw);
175 
176  ld = gw->g.width < gw->g.height ? gw->g.width : gw->g.height;
177 
178  #if GDISP_NEED_CIRCLE
179  df = (ld-1)/2;
180  gdispFillArea(gw->g.x, gw->g.y, ld, ld, gw->pstyle->background);
181  gdispDrawCircle(gw->g.x+df, gw->g.y+df, df, pcol->edge);
182 
183  if (gw->g.flags & GRADIO_FLG_PRESSED)
184  gdispFillCircle(gw->g.x+df, gw->g.y+df, df <= 2 ? 1 : (df-2), pcol->fill);
185  #else
186  gdispFillArea(gw->g.x+1, gw->g.y+1, ld, ld-2, gw->pstyle->background);
187  gdispDrawBox(gw->g.x, gw->g.y, ld, ld, pcol->edge);
188 
189  df = ld < 4 ? 1 : 2;
190  if (gw->g.flags & GRADIO_FLG_PRESSED)
191  gdispFillArea(gw->g.x+df, gw->g.y+df, ld-2*df, ld-2*df, pcol->fill);
192  #endif
193 
194  gdispFillStringBox(gw->g.x+ld+1, gw->g.y, gw->g.width-ld-1, gw->g.height, gw->text, gw->g.font, pcol->text, gw->pstyle->background, justifyLeft);
195  #undef gcw
196 }
