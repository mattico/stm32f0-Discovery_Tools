40 static void pointto(GGraphObject *gg, coord_t x, coord_t y, const GGraphPointStyle *style) {
41  if (style->type == GGRAPH_POINT_NONE)
42  return;
43 
44  // Convert to device space. Note the y-axis is inverted.
45  x += gg->g.x + gg->xorigin;
46  y = gg->g.y + gg->g.height - 1 - gg->yorigin - y;
47 
48  if (style->size <= 1) {
49  gdispDrawPixel(x, y, style->color);
50  return;
51  }
52 
53  switch(style->type) {
54  case GGRAPH_POINT_SQUARE:
55  gdispDrawBox(x-style->size, y-style->size, 2*style->size, 2*style->size, style->color);
56  break;
57 #if GDISP_NEED_CIRCLE
58  case GGRAPH_POINT_CIRCLE:
59  gdispDrawCircle(x, y, style->size, style->color);
60  break;
61 #endif
62  case GGRAPH_POINT_DOT:
63  default:
64  gdispDrawPixel(x, y, style->color);
65  break;
66  }
67 }

69 static void lineto(GGraphObject *gg, coord_t x0, coord_t y0, coord_t x1, coord_t y1, const GGraphLineStyle *style) {
70  coord_t dy, dx;
71  coord_t addx, addy;
72  coord_t P, diff, i;
73  coord_t run_on, run_off, run;
74 
75  if (style->type == GGRAPH_LINE_NONE)
76  return;
77 
78  // Convert to device space. Note the y-axis is inverted.
79  x0 += gg->g.x + gg->xorigin;
80  y0 = gg->g.y + gg->g.height - 1 - gg->yorigin - y0;
81  x1 += gg->g.x + gg->xorigin;
82  y1 = gg->g.y + gg->g.height - 1 - gg->yorigin - y1;
83 
84  if (style->size <= 0) {
85  // Use the driver to draw a solid line
86  gdispDrawLine(x0, y0, x1, y1, style->color);
87  return;
88  }
89 
90  switch (style->type) {
91  case GGRAPH_LINE_DOT:
92  run_on = 1;
93  run_off = -style->size;
94  break;
95 
96  case GGRAPH_LINE_DASH:
97  run_on = style->size;
98  run_off = -style->size;
99  break;
100 
101  case GGRAPH_LINE_SOLID:
102  default:
103  // Use the driver to draw a solid line
104  gdispDrawLine(x0, y0, x1, y1, style->color);
105  return;
106  }
107 
108  // Use Bresenham's algorithm modified to draw a stylized line
109  run = 0;
110  if (x1 >= x0) {
111  dx = x1 - x0;
112  addx = 1;
113  } else {
114  dx = x0 - x1;
115  addx = -1;
116  }
117  if (y1 >= y0) {
118  dy = y1 - y0;
119  addy = 1;
120  } else {
121  dy = y0 - y1;
122  addy = -1;
123  }
124 
125  if (dx >= dy) {
126  dy *= 2;
127  P = dy - dx;
128  diff = P - dx;
129 
130  for(i=0; i<=dx; ++i) {
131  if (run++ >= 0) {
132  if (run >= run_on)
133  run = run_off;
134  gdispDrawPixel(x0, y0, style->color);
135  }
136  if (P < 0) {
137  P += dy;
138  x0 += addx;
139  } else {
140  P += diff;
141  x0 += addx;
142  y0 += addy;
143  }
144  }
145  } else {
146  dx *= 2;
147  P = dx - dy;
148  diff = P - dy;
149 
150  for(i=0; i<=dy; ++i) {
151  if (run++ >= 0) {
152  if (run >= run_on)
153  run = run_off;
154  gdispDrawPixel(x0, y0, style->color);
155  }
156  if (P < 0) {
157  P += dx;
158  y0 += addy;
159  } else {
160  P += diff;
161  x0 += addx;
162  y0 += addy;
163  }
164  }
165  }
166 }

207 void gwinGraphDrawAxis(GHandle gh) {
208  #define gg ((GGraphObject *)gh)
209  coord_t i, xmin, ymin, xmax, ymax;
210 
211  if (gh->vmt != &graphVMT)
212  return;
213 
214  xmin = -gg->xorigin;
215  xmax = gh->width-gg->xorigin-1;
216  ymin = -gg->yorigin;
217  ymax = gh->height-gg->yorigin-1;
218 
219  // x grid - this code assumes that the GGraphGridStyle is a superset of GGraphListStyle
220  if (gg->style.xgrid.type != GGRAPH_LINE_NONE && gg->style.xgrid.spacing >= 2) {
221  for(i = gg->style.xgrid.spacing; i <= xmax; i += gg->style.xgrid.spacing)
222  lineto(gg, i, ymin, i, ymax, (GGraphLineStyle *)&gg->style.xgrid);
223  for(i = -gg->style.xgrid.spacing; i >= xmin; i -= gg->style.xgrid.spacing)
224  lineto(gg, i, ymin, i, ymax, (GGraphLineStyle *)&gg->style.xgrid);
225  }
226 
227  // y grid - this code assumes that the GGraphGridStyle is a superset of GGraphListStyle
228  if (gg->style.ygrid.type != GGRAPH_LINE_NONE && gg->style.ygrid.spacing >= 2) {
229  for(i = gg->style.ygrid.spacing; i <= ymax; i += gg->style.ygrid.spacing)
230  lineto(gg, xmin, i, xmax, i, (GGraphLineStyle *)&gg->style.ygrid);
231  for(i = -gg->style.ygrid.spacing; i >= ymin; i -= gg->style.ygrid.spacing)
232  lineto(gg, xmin, i, xmax, i, (GGraphLineStyle *)&gg->style.ygrid);
233  }
234 
235  // x axis
236  lineto(gg, xmin, 0, xmax, 0, &gg->style.xaxis);
237  if ((gg->style.flags & GWIN_GRAPH_STYLE_XAXIS_NEGATIVE_ARROWS)) {
238  if (xmin > 0 || xmin < -(GGRAPH_ARROW_SIZE+1)) {
239  lineto(gg, xmin, 0, xmin+GGRAPH_ARROW_SIZE, GGRAPH_ARROW_SIZE, &gg->style.xaxis);
240  lineto(gg, xmin, 0, xmin+GGRAPH_ARROW_SIZE, -GGRAPH_ARROW_SIZE, &gg->style.xaxis);
241  }
242  }
243  if ((gg->style.flags & GWIN_GRAPH_STYLE_XAXIS_POSITIVE_ARROWS)) {
244  if (xmax < 0 || xmax > (GGRAPH_ARROW_SIZE+1)) {
245  lineto(gg, xmax, 0, xmax-GGRAPH_ARROW_SIZE, GGRAPH_ARROW_SIZE, &gg->style.xaxis);
246  lineto(gg, xmax, 0, xmax-GGRAPH_ARROW_SIZE, -GGRAPH_ARROW_SIZE, &gg->style.xaxis);
247  }
248  }
249 
250  // y axis
251  lineto(gg, 0, ymin, 0, ymax, &gg->style.yaxis);
252  if ((gg->style.flags & GWIN_GRAPH_STYLE_YAXIS_NEGATIVE_ARROWS)) {
253  if (ymin > 0 || ymin < -(GGRAPH_ARROW_SIZE+1)) {
254  lineto(gg, 0, ymin, GGRAPH_ARROW_SIZE, ymin+GGRAPH_ARROW_SIZE, &gg->style.yaxis);
255  lineto(gg, 0, ymin, -GGRAPH_ARROW_SIZE, ymin+GGRAPH_ARROW_SIZE, &gg->style.yaxis);
256  }
257  }
258  if ((gg->style.flags & GWIN_GRAPH_STYLE_YAXIS_POSITIVE_ARROWS)) {
259  if (ymax < 0 || ymax > (GGRAPH_ARROW_SIZE+1)) {
260  lineto(gg, 0, ymax, GGRAPH_ARROW_SIZE, ymax-GGRAPH_ARROW_SIZE, &gg->style.yaxis);
261  lineto(gg, 0, ymax, -GGRAPH_ARROW_SIZE, ymax-GGRAPH_ARROW_SIZE, &gg->style.yaxis);
262  }
263  }
264 
265  #undef gg
266 }

275 void gwinGraphDrawPoint(GHandle gh, coord_t x, coord_t y) {
276  #define gg ((GGraphObject *)gh)
277 
278  if (gh->vmt != &graphVMT)
279  return;
280 
281  if ((gh->flags & GGRAPH_FLG_CONNECTPOINTS)) {
282  // Draw the line
283  lineto(gg, gg->lastx, gg->lasty, x, y, &gg->style.line);
284 
285  // Redraw the previous point because the line may have overwritten it
286  pointto(gg, gg->lastx, gg->lasty, &gg->style.point);
287 
288  } else
289  gh->flags |= GGRAPH_FLG_CONNECTPOINTS;
290 
291  // Save this point for next time.
292  gg->lastx = x;
293  gg->lasty = y;
294 
295  // Draw this point.
296  pointto(gg, x, y, &gg->style.point);
297 
298  #undef gg
299 }

301 void gwinGraphDrawPoints(GHandle gh, const point *points, unsigned count) {
302  #define gg ((GGraphObject *)gh)
303  unsigned i;
304  const point *p;
305 
306  if (gh->vmt != &graphVMT)
307  return;
308 
309  // Draw the connecting lines
310  for(p = points, i = 0; i < count; p++, i++) {
311  if ((gh->flags & GGRAPH_FLG_CONNECTPOINTS)) {
312  // Draw the line
313  lineto(gg, gg->lastx, gg->lasty, p->x, p->y, &gg->style.line);
314 
315  // Redraw the previous point because the line may have overwritten it
316  if (i == 0)
317  pointto(gg, gg->lastx, gg->lasty, &gg->style.point);
318 
319  } else
320  gh->flags |= GGRAPH_FLG_CONNECTPOINTS;
321 
322  // Save this point for next time.
323  gg->lastx = p->x;
324  gg->lasty = p->y;
325  }
326 
327 
328  // Draw the points.
329  for(p = points, i = 0; i < count; p++, i++)
330  pointto(gg, p->x, p->y, &gg->style.point);
331 
332  #undef gg
333 }
