84 static void gwidgetEvent(void *param, GEvent *pe) {
85  #define gh QItem2GWindow(qi)
86  #define pme ((GEventMouse *)pe)
87  #define pte ((GEventToggle *)pe)
88  #define pde ((GEventDial *)pe)
89 
90  const gfxQueueASyncItem * qi;
91  #if GFX_USE_GINPUT && (GINPUT_NEED_TOGGLE || GINPUT_NEED_DIAL)
92  uint16_t role;
93  #endif
94  (void) param;
95 
96  // Process various events
97  switch (pe->type) {
98 
99  #if GFX_USE_GINPUT && GINPUT_NEED_MOUSE
100  case GEVENT_MOUSE:
101  case GEVENT_TOUCH:
102  // Cycle through all windows
103  for(qi = gfxQueueASyncPeek(&_GWINList); qi; qi = gfxQueueASyncNext(qi)) {
104 
105  // check if it a widget that is enabled and visible
106  if ((gh->flags & (GWIN_FLG_WIDGET|GWIN_FLG_ENABLED|GWIN_FLG_VISIBLE)) != (GWIN_FLG_WIDGET|GWIN_FLG_ENABLED|GWIN_FLG_VISIBLE))
107  continue;
108 
109  // Are we captured?
110  if ((gw->g.flags & GWIN_FLG_MOUSECAPTURE)) {
111  if ((pme->last_buttons & ~pme->current_buttons & GINPUT_MOUSE_BTN_LEFT)) {
112  gw->g.flags &= ~GWIN_FLG_MOUSECAPTURE;
113  if (wvmt->MouseUp)
114  wvmt->MouseUp(gw, pme->x - gw->g.x, pme->y - gw->g.y);
115  } else if (wvmt->MouseMove)
116  wvmt->MouseMove(gw, pme->x - gw->g.x, pme->y - gw->g.y);
117 
118  // We are not captured - look for mouse downs over the widget
119  } else if ((~pme->last_buttons & pme->current_buttons & GINPUT_MOUSE_BTN_LEFT)
120  && pme->x >= gw->g.x && pme->x < gw->g.x + gw->g.width
121  && pme->y >= gw->g.y && pme->y < gw->g.y + gw->g.height) {
122  gw->g.flags |= GWIN_FLG_MOUSECAPTURE;
123  if (wvmt->MouseDown)
124  wvmt->MouseDown(gw, pme->x - gw->g.x, pme->y - gw->g.y);
125  }
126  }
127  break;
128  #endif
129 
130  #if GFX_USE_GINPUT && GINPUT_NEED_TOGGLE
131  case GEVENT_TOGGLE:
132  // Cycle through all windows
133  for(qi = gfxQueueASyncPeek(&_GWINList); qi; qi = gfxQueueASyncNext(qi)) {
134 
135  // check if it a widget that is enabled and visible
136  if ((gh->flags & (GWIN_FLG_WIDGET|GWIN_FLG_ENABLED|GWIN_FLG_VISIBLE)) != (GWIN_FLG_WIDGET|GWIN_FLG_ENABLED|GWIN_FLG_VISIBLE))
137  continue;
138 
139  for(role = 0; role < wvmt->toggleroles; role++) {
140  if (wvmt->ToggleGet(gw, role) == pte->instance) {
141  if (pte->on) {
142  if (wvmt->ToggleOn)
143  wvmt->ToggleOn(gw, role);
144  } else {
145  if (wvmt->ToggleOff)
146  wvmt->ToggleOff(gw, role);
147  }
148  }
149  }
150  }
151  break;
152  #endif
153 
154  #if GFX_USE_GINPUT && GINPUT_NEED_DIAL
155  case GEVENT_DIAL:
156  // Cycle through all windows
157  for(qi = gfxQueueASyncPeek(&_GWINList); qi; qi = gfxQueueASyncNext(qi)) {
158 
159  // check if it a widget that is enabled and visible
160  if ((gh->flags & (GWIN_FLG_WIDGET|GWIN_FLG_ENABLED|GWIN_FLG_VISIBLE)) != (GWIN_FLG_WIDGET|GWIN_FLG_ENABLED|GWIN_FLG_VISIBLE))
161  continue;
162 
163  for(role = 0; role < wvmt->dialroles; role++) {
164  if (wvmt->DialGet(gw, role) == pte->instance) {
165  if (wvmt->DialMove)
166  wvmt->DialMove(gw, role, pde->value, pde->maxvalue);
167  }
168  }
169  }
170  break;
171  #endif
172 
173  default:
174  break;
175  }
176 
177  #undef gh
178  #undef pme
179  #undef pte
180  #undef pde
181 }

240 void _gwidgetDestroy(GHandle gh) {
241  #if GFX_USE_GINPUT && (GINPUT_NEED_TOGGLE || GINPUT_NEED_DIAL)
242  uint16_t role, instance;
243  #endif
244 
245  // Deallocate the text (if necessary)
246  if ((gh->flags & GWIN_FLG_ALLOCTXT)) {
247  gh->flags &= ~GWIN_FLG_ALLOCTXT;
248  gfxFree((void *)gw->text);
249  }
250 
251  #if GFX_USE_GINPUT && GINPUT_NEED_TOGGLE
252  // Detach any toggles from this object
253  for(role = 0; role < wvmt->toggleroles; role++) {
254  instance = wvmt->ToggleGet(gw, role);
255  if (instance != GWIDGET_NO_INSTANCE) {
256  wvmt->ToggleAssign(gw, role, GWIDGET_NO_INSTANCE);
257  if (!FindToggleUser(instance))
258  geventDetachSource(&gl, ginputGetToggle(instance));
259  }
260  }
261  #endif
262 
263  #if GFX_USE_GINPUT && GINPUT_NEED_DIAL
264  // Detach any dials from this object
265  for(role = 0; role < wvmt->dialroles; role++) {
266  instance = wvmt->DialGet(gw, role);
267  if (instance != GWIDGET_NO_INSTANCE) {
268  wvmt->DialAssign(gw, role, GWIDGET_NO_INSTANCE);
269  if (!FindDialUser(instance))
270  geventDetachSource(&gl, ginputGetDial(instance));
271  }
272  }
273  #endif
274 
275  // Remove any listeners on this object.
276  geventDetachSourceListeners((GSourceHandle)gh);
277 }

318 void gwinSetText(GHandle gh, const char *text, bool_t useAlloc) {
319  if (!(gh->flags & GWIN_FLG_WIDGET))
320  return;
321 
322  // Dispose of the old string
323  if ((gh->flags & GWIN_FLG_ALLOCTXT)) {
324  gh->flags &= ~GWIN_FLG_ALLOCTXT;
325  if (gw->text) {
326  gfxFree((void *)gw->text);
327  gw->text = "";
328  }
329  }
330 
331  // Alloc the new text if required
332  if (!text || !*text)
333  gw->text = "";
334  else if (useAlloc) {
335  char *str;
336 
337  if ((str = (char *)gfxAlloc(strlen(text)+1))) {
338  gh->flags |= GWIN_FLG_ALLOCTXT;
339  strcpy(str, text);
340  }
341  gw->text = (const char *)str;
342  } else
343  gw->text = text;
344  _gwidgetRedraw(gh);
345 }

391 #if GFX_USE_GINPUT && GINPUT_NEED_TOGGLE
392  bool_t gwinAttachToggle(GHandle gh, uint16_t role, uint16_t instance) {
393  GSourceHandle gsh;
394  uint16_t oi;
395 
396  // Is this a widget
397  if (!(gh->flags & GWIN_FLG_WIDGET))
398  return FALSE;
399 
400  // Is the role valid
401  if (role >= wvmt->toggleroles)
402  return FALSE;
403 
404  // Is this a valid device
405  if (!(gsh = ginputGetToggle(instance)))
406  return FALSE;
407 
408  // Is this already done?
409  oi = wvmt->ToggleGet(gw, role);
410  if (instance == oi)
411  return TRUE;
412 
413  // Remove the old instance
414  if (oi != GWIDGET_NO_INSTANCE) {
415  wvmt->ToggleAssign(gw, role, GWIDGET_NO_INSTANCE);
416  if (!FindToggleUser(oi))
417  geventDetachSource(&gl, ginputGetToggle(oi));
418  }
419 
420  // Assign the new
421  wvmt->ToggleAssign(gw, role, instance);
422  return geventAttachSource(&gl, gsh, GLISTEN_TOGGLE_ON|GLISTEN_TOGGLE_OFF);
423  }
424 #endif
