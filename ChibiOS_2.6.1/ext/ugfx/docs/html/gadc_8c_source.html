85 /* Find the next conversion to activate */
86 static inline void FindNextConversionI(void) {
87  if (curlsdev) {
88  /**
89  * Now we have done a low speed conversion - start looking for the next conversion
90  * We only look forward to ensure we get a high speed conversion at least once
91  * every GADC_MAX_LOWSPEED_DEVICES conversions.
92  */
93  curlsdev++;
94 
95  } else {
96 
97  /* Now we have done a high speed conversion - start looking for low speed conversions */
98  curlsdev = ls;
99  }
100 
101  /**
102  * Look for the next thing to do.
103  */
104  while(curlsdev < &ls[GADC_MAX_LOWSPEED_DEVICES]) {
105  if ((curlsdev->flags & (GADC_FLG_ISACTIVE|GADC_FLG_ISDONE)) == GADC_FLG_ISACTIVE) {
106  gadc_lld_adc_nontimerI(&curlsdev->lld);
107  return;
108  }
109  curlsdev++;
110  }
111  curlsdev = 0;
112 
113  /* No more low speed devices - do a high speed conversion */
114  if (hs.flags & GADC_FLG_ISACTIVE) {
115  hs.lld.now = GADC_Timer_Missed ? TRUE : FALSE;
116  GADC_Timer_Missed = 0;
117  gadc_lld_adc_timerI(&hs.lld);
118  return;
119  }
120 
121  /* Nothing more to do */
122  gflags &= ~GADC_GFLG_ISACTIVE;
123 }

125 void GADC_ISR_CompleteI(ADCDriver *adcp, adcsample_t *buffer, size_t n) {
126  (void) adcp;
127 
128  if (curlsdev) {
129  /* This interrupt must be in relation to the low speed device */
130 
131  if (curlsdev->flags & GADC_FLG_ISACTIVE) {
132  /**
133  * As we only handle a single low speed conversion at a time, we know
134  * we know we won't get any half completion interrupts.
135  */
136  curlsdev->flags |= GADC_FLG_ISDONE;
137  gtimerJabI(&LowSpeedGTimer);
138  }
139 
140  #if ADC_ISR_FULL_CODE_BUG
141  /**
142  * Oops - We have just finished a low speed conversion but a bug prevents us
143  * restarting the ADC here. Other code will restart it in the thread based
144  * ADC handler.
145  */
146  gflags &= ~GADC_GFLG_ISACTIVE;
147  return;
148 
149  #endif
150 
151  } else {
152  /* This interrupt must be in relation to the high speed device */
153 
154  if (hs.flags & GADC_FLG_ISACTIVE) {
155  /* Save the details */
156  hs.lastcount = n;
157  hs.lastbuffer = buffer;
158  hs.lastflags = GADC_Timer_Missed ? GADC_HSADC_LOSTEVENT : 0;
159 
160  /* Signal the user with the data */
161  if (hs.pEvent) {
162  #if GFX_USE_GEVENT
163  hs.pEvent->type = GEVENT_ADC;
164  #endif
165  hs.pEvent->count = hs.lastcount;
166  hs.pEvent->buffer = hs.lastbuffer;
167  hs.pEvent->flags = hs.lastflags;
168  }
169 
170  /* Our three signalling mechanisms */
171  if (hs.isrfn)
172  hs.isrfn(buffer, n);
173 
174  if (hs.bsem)
175  gfxSemSignalI(hs.bsem);
176 
177  #if GFX_USE_GEVENT
178  if (hs.flags & GADC_FLG_GTIMER)
179  gtimerJabI(&HighSpeedGTimer);
180  #endif
181 
182  /* Adjust what we have left to do */
183  hs.lld.count -= n;
184  hs.remaining -= n;
185 
186  /* Half completion - We have done all we can for now - wait for the next interrupt */
187  if (hs.lld.count)
188  return;
189 
190  /* Our buffer is cyclic - set up the new buffer pointers */
191  if (hs.remaining) {
192  hs.lld.buffer = buffer + (n * hs.samplesPerConversion);
193  } else {
194  hs.remaining = hs.bufcount;
195  hs.lld.buffer = hs.buffer;
196  }
197  hs.lld.count = hs.remaining < hs.samplesPerEvent ? hs.remaining : hs.samplesPerEvent;
198  }
199  }
200 
201  /**
202  * Look for the next thing to do.
203  */
204  FindNextConversionI();
205 }

207 void GADC_ISR_ErrorI(ADCDriver *adcp, adcerror_t err) {
208  (void) adcp;
209  (void) err;
210 
211  if (curlsdev) {
212  if ((curlsdev->flags & (GADC_FLG_ISACTIVE|GADC_FLG_ISDONE)) == GADC_FLG_ISACTIVE)
213  /* Mark the error then try to repeat it */
214  curlsdev->flags |= GADC_FLG_ERROR;
215 
216  #if ADC_ISR_FULL_CODE_BUG
217  /**
218  * Oops - We have just finished a low speed conversion but a bug prevents us
219  * restarting the ADC here. Other code will restart it in the thread based
220  * ADC handler.
221  */
222  gflags &= ~GADC_GFLG_ISACTIVE;
223  gtimerJabI(&LowSpeedGTimer);
224  return;
225 
226  #endif
227 
228  } else {
229  if (hs.flags & GADC_FLG_ISACTIVE)
230  /* Mark the error and then try to repeat it */
231  hs.flags |= GADC_FLG_ERROR;
232  }
233 
234  /* Start the next conversion */
235  FindNextConversionI();
236 }

263 #if GFX_USE_GEVENT
264  static void HighSpeedGTimerCallback(void *param) {
265  (void) param;
266  GSourceListener *psl;
267  GEventADC *pe;
268 
269  psl = 0;
270  while ((psl = geventGetSourceListener((GSourceHandle)(&HighSpeedGTimer), psl))) {
271  if (!(pe = (GEventADC *)geventGetEventBuffer(psl))) {
272  // This listener is missing - save this.
273  psl->srcflags |= GADC_HSADC_LOSTEVENT;
274  continue;
275  }
276 
277  pe->type = GEVENT_ADC;
278  pe->count = hs.lastcount;
279  pe->buffer = hs.lastbuffer;
280  pe->flags = hs.lastflags | psl->srcflags;
281  psl->srcflags = 0;
282  geventSendEvent(psl);
283  }
284  }
285 #endif

287 static void LowSpeedGTimerCallback(void *param) {
288  (void) param;
289  GADCCallbackFunction fn;
290  void *prm;
291  adcsample_t *buffer;
292  struct lsdev *p;
293 
294  #if ADC_ISR_FULL_CODE_BUG
295  /* Ensure the ADC is running if it needs to be - Bugfix HACK */
296  StartADC(FALSE);
297  #endif
298 
299  /**
300  * Look for completed low speed timers.
301  * We don't need to take the mutex as we are the only place that things are freed and we
302  * do that atomically.
303  */
304  for(p=ls; p < &ls[GADC_MAX_LOWSPEED_DEVICES]; p++) {
305  if ((p->flags & (GADC_FLG_ISACTIVE|GADC_FLG_ISDONE)) == (GADC_FLG_ISACTIVE|GADC_FLG_ISDONE)) {
306  /* This item is done - perform its callback */
307  fn = p->fn; // Save the callback details
308  prm = p->param;
309  buffer = p->lld.buffer;
310  p->fn = 0; // Needed to prevent the compiler removing the local variables
311  p->param = 0; // Needed to prevent the compiler removing the local variables
312  p->lld.buffer = 0; // Needed to prevent the compiler removing the local variables
313  p->flags = 0; // The slot is available (indivisible operation)
314  gfxSemSignal(&gadcsem); // Tell everyone
315  fn(buffer, prm); // Perform the callback
316  }
317  }

388 void gadcLowSpeedGet(uint32_t physdev, adcsample_t *buffer) {
389  struct lsdev *p;
390  gfxSem mysem;
391 
392  /* Start the Low Speed Timer */
393  gfxSemInit(&mysem, 1, 1);
394  gfxMutexEnter(&gadcmutex);
395  if (!gtimerIsActive(&LowSpeedGTimer))
396  gtimerStart(&LowSpeedGTimer, LowSpeedGTimerCallback, NULL, TRUE, TIME_INFINITE);
397  gfxMutexExit(&gadcmutex);
398 
399  while(1) {
400  /* Wait for an available slot */
401  gfxSemWait(&gadcsem, TIME_INFINITE);
402 
403  /* Find a slot */
404  gfxMutexEnter(&gadcmutex);
405  for(p = ls; p < &ls[GADC_MAX_LOWSPEED_DEVICES]; p++) {
406  if (!(p->flags & GADC_FLG_ISACTIVE)) {
407  p->lld.physdev = physdev;
408  p->lld.buffer = buffer;
409  p->fn = BSemSignalCallback;
410  p->param = &mysem;
411  p->flags = GADC_FLG_ISACTIVE;
412  gfxMutexExit(&gadcmutex);
413  StartADC(FALSE);
414  gfxSemWait(&mysem, TIME_INFINITE);
415  return;
416  }
417  }
418  gfxMutexExit(&gadcmutex);
419 
420  /**
421  * We should never get here - the count semaphore must be wrong.
422  * Decrement it and try again.
423  */
424  }
425 }

427 bool_t gadcLowSpeedStart(uint32_t physdev, adcsample_t *buffer, GADCCallbackFunction fn, void *param) {
428  struct lsdev *p;
429 
430  /* Start the Low Speed Timer */
431  gfxMutexEnter(&gadcmutex);
432  if (!gtimerIsActive(&LowSpeedGTimer))
433  gtimerStart(&LowSpeedGTimer, LowSpeedGTimerCallback, NULL, TRUE, TIME_INFINITE);
434 
435  /* Find a slot */
436  for(p = ls; p < &ls[GADC_MAX_LOWSPEED_DEVICES]; p++) {
437  if (!(p->flags & GADC_FLG_ISACTIVE)) {
438  /* We know we have a slot - this should never wait anyway */
439  gfxSemWait(&gadcsem, TIME_IMMEDIATE);
440  p->lld.physdev = physdev;
441  p->lld.buffer = buffer;
442  p->fn = fn;
443  p->param = param;
444  p->flags = GADC_FLG_ISACTIVE;
445  gfxMutexExit(&gadcmutex);
446  StartADC(FALSE);
447  return TRUE;
448  }
449  }
450  gfxMutexExit(&gadcmutex);
451  return FALSE;
452 }
