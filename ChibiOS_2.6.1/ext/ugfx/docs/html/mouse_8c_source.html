104  static inline void _tsDo3PointCalibration(const MousePoint *cross, const MousePoint *points, Calibration *c) {
105  float dx, dx0, dx1, dx2, dy0, dy1, dy2;
106 
107  /* Compute all the required determinants */
108  dx = ((float)(points[0].x - points[2].x)) * ((float)(points[1].y - points[2].y))
109  - ((float)(points[1].x - points[2].x)) * ((float)(points[0].y - points[2].y));
110 
111  dx0 = ((float)(cross[0].x - cross[2].x)) * ((float)(points[1].y - points[2].y))
112  - ((float)(cross[1].x - cross[2].x)) * ((float)(points[0].y - points[2].y));
113 
114  dx1 = ((float)(cross[1].x - cross[2].x)) * ((float)(points[0].x - points[2].x))
115  - ((float)(cross[0].x - cross[2].x)) * ((float)(points[1].x - points[2].x));
116 
117  dx2 = cross[0].x * ((float)points[1].x * (float)points[2].y - (float)points[2].x * (float)points[1].y) -
118  cross[1].x * ((float)points[0].x * (float)points[2].y - (float)points[2].x * (float)points[0].y) +
119  cross[2].x * ((float)points[0].x * (float)points[1].y - (float)points[1].x * (float)points[0].y);
120 
121  dy0 = ((float)(cross[0].y - cross[2].y)) * ((float)(points[1].y - points[2].y))
122  - ((float)(cross[1].y - cross[2].y)) * ((float)(points[0].y - points[2].y));
123 
124  dy1 = ((float)(cross[1].y - cross[2].y)) * ((float)(points[0].x - points[2].x))
125  - ((float)(cross[0].y - cross[2].y)) * ((float)(points[1].x - points[2].x));
126 
127  dy2 = cross[0].y * ((float)points[1].x * (float)points[2].y - (float)points[2].x * (float)points[1].y) -
128  cross[1].y * ((float)points[0].x * (float)points[2].y - (float)points[2].x * (float)points[0].y) +
129  cross[2].y * ((float)points[0].x * (float)points[1].y - (float)points[1].x * (float)points[0].y);
130 
131  /* Now, calculate all the required coefficients */
132  c->ax = dx0 / dx;
133  c->bx = dx1 / dx;
134  c->cx = dx2 / dx;
135 
136  c->ay = dy0 / dx;
137  c->by = dy1 / dx;
138  c->cy = dy2 / dx;
139  }

164 static void get_calibrated_reading(MouseReading *pt) {
165  #if GINPUT_MOUSE_NEED_CALIBRATION || GDISP_NEED_CONTROL
166  coord_t w, h;
167  #endif
168 
169  get_raw_reading(pt);
170 
171  #if GINPUT_MOUSE_NEED_CALIBRATION || GDISP_NEED_CONTROL
172  w = gdispGetWidth();
173  h = gdispGetHeight();
174  #endif
175 
176  #if GINPUT_MOUSE_NEED_CALIBRATION
177  _tsTransform(pt, &MouseConfig.caldata);
178  #endif
179 
180  #if GDISP_NEED_CONTROL
181  switch(gdispGetOrientation()) {
182  case GDISP_ROTATE_0:
183  break;
184  case GDISP_ROTATE_90:
185  {
186  coord_t t = pt->y;
187  pt->y = h - 1 - pt->x;
188  pt->x = t;
189  }
190  break;
191  case GDISP_ROTATE_180:
192  pt->x = w - 1 - pt->x;
193  pt->y = h - 1 - pt->y;
194  break;
195  case GDISP_ROTATE_270:
196  {
197  coord_t t = pt->x;
198  pt->x = w - 1 - pt->y;
199  pt->y = t;
200  }
201  break;
202  }
203  #endif
204 
205  #if GINPUT_MOUSE_NEED_CALIBRATION
206  if (pt->x < 0) pt->x = 0;
207  else if (pt->x >= w) pt->x = w-1;
208  if (pt->y < 0) pt->y = 0;
209  else if (pt->y >= h) pt->y = h-1;
210  #endif
211 }

213 static void MousePoll(void *param) {
214  (void) param;
215  GSourceListener *psl;
216  GEventMouse *pe;
217  unsigned meta;
218  uint16_t tbtns;
219  uint32_t cdiff;
220  uint32_t mdiff;
221 
222  // Save the last mouse state
223  MouseConfig.last_buttons = MouseConfig.t.buttons;
224 
225  // Get the new mouse reading
226  get_calibrated_reading(&MouseConfig.t);
227 
228  // Calculate out new event meta value and handle CLICK and CXTCLICK
229  meta = GMETA_NONE;
230 
231  // Calculate the position difference from our movement reference (update the reference if out of range)
232  mdiff = (MouseConfig.t.x - MouseConfig.movepos.x) * (MouseConfig.t.x - MouseConfig.movepos.x) +
233  (MouseConfig.t.y - MouseConfig.movepos.y) * (MouseConfig.t.y - MouseConfig.movepos.y);
234  if (mdiff > GINPUT_MOUSE_MAX_MOVE_JITTER * GINPUT_MOUSE_MAX_MOVE_JITTER) {
235  MouseConfig.movepos.x = MouseConfig.t.x;
236  MouseConfig.movepos.y = MouseConfig.t.y;
237  }
238 
239  // Check if the click has moved outside the click area and if so cancel the click
240  if ((MouseConfig.flags & FLG_CLICK_TIMER)) {
241  cdiff = (MouseConfig.t.x - MouseConfig.clickpos.x) * (MouseConfig.t.x - MouseConfig.clickpos.x) +
242  (MouseConfig.t.y - MouseConfig.clickpos.y) * (MouseConfig.t.y - MouseConfig.clickpos.y);
243  if (cdiff > GINPUT_MOUSE_MAX_CLICK_JITTER * GINPUT_MOUSE_MAX_CLICK_JITTER)
244  MouseConfig.flags &= ~FLG_CLICK_TIMER;
245  }
246 
247  // Mouse down
248  tbtns = MouseConfig.t.buttons & ~MouseConfig.last_buttons;
249  if ((tbtns & GINPUT_MOUSE_BTN_LEFT))
250  meta |= GMETA_MOUSE_DOWN;
251  if ((tbtns & (GINPUT_MOUSE_BTN_LEFT|GINPUT_MOUSE_BTN_RIGHT))) {
252  MouseConfig.clickpos.x = MouseConfig.t.x;
253  MouseConfig.clickpos.y = MouseConfig.t.y;
254  MouseConfig.clicktime = gfxSystemTicks();
255  MouseConfig.flags |= FLG_CLICK_TIMER;
256  }
257 
258  // Mouse up
259  tbtns = ~MouseConfig.t.buttons & MouseConfig.last_buttons;
260  if ((tbtns & GINPUT_MOUSE_BTN_LEFT))
261  meta |= GMETA_MOUSE_UP;
262  if ((tbtns & (GINPUT_MOUSE_BTN_LEFT|GINPUT_MOUSE_BTN_RIGHT))) {
263  if ((MouseConfig.flags & FLG_CLICK_TIMER)) {
264  if ((tbtns & GINPUT_MOUSE_BTN_LEFT)
265  #if GINPUT_MOUSE_CLICK_TIME != TIME_INFINITE
266  && gfxSystemTicks() - MouseConfig.clicktime < gfxMillisecondsToTicks(GINPUT_MOUSE_CLICK_TIME)
267  #endif
268  )
269  meta |= GMETA_MOUSE_CLICK;
270  else
271  meta |= GMETA_MOUSE_CXTCLICK;
272  MouseConfig.flags &= ~FLG_CLICK_TIMER;
273  }
274  }
275 
276  // Send the event to the listeners that are interested.
277  psl = 0;
278  while ((psl = geventGetSourceListener((GSourceHandle)(&MouseConfig), psl))) {
279  if (!(pe = (GEventMouse *)geventGetEventBuffer(psl))) {
280  // This listener is missing - save the meta events that have happened
281  psl->srcflags |= meta;
282  continue;
283  }
284 
285  // If we haven't really moved (and there are no meta events) don't bother sending the event
286  if (mdiff <= GINPUT_MOUSE_MAX_MOVE_JITTER * GINPUT_MOUSE_MAX_MOVE_JITTER && !psl->srcflags && !meta && !(psl->listenflags & GLISTEN_MOUSENOFILTER))
287  continue;
288 
289  // Send the event if we are listening for it
290  if (((MouseConfig.t.buttons & GINPUT_MOUSE_BTN_LEFT) && (psl->listenflags & GLISTEN_MOUSEDOWNMOVES))
291  || (!(MouseConfig.t.buttons & GINPUT_MOUSE_BTN_LEFT) && (psl->listenflags & GLISTEN_MOUSEUPMOVES))
292  || (meta && (psl->listenflags & GLISTEN_MOUSEMETA))) {
293  pe->type = GINPUT_MOUSE_EVENT_TYPE;
294  pe->instance = 0;
295  pe->x = MouseConfig.t.x;
296  pe->y = MouseConfig.t.y;
297  pe->z = MouseConfig.t.z;
298  pe->current_buttons = MouseConfig.t.buttons;
299  pe->last_buttons = MouseConfig.last_buttons;
300  pe->meta = meta;
301  if (psl->srcflags) {
302  pe->current_buttons |= GINPUT_MISSED_MOUSE_EVENT;
303  pe->meta |= psl->srcflags;
304  psl->srcflags = 0;
305  }
306  geventSendEvent(psl);
307  }
308  }

311 GSourceHandle ginputGetMouse(uint16_t instance) {
312  #if GINPUT_MOUSE_NEED_CALIBRATION
313  Calibration *pc;
314  #endif
315 
316  // We only support a single mouse instance currently
317  // Instance 9999 is the same as instance 0 except that it installs
318  // a special "raw" calibration if there isn't one we can load.
319  if (instance && instance != 9999)
320  return 0;
321 
322  // Do we need to initialise the mouse subsystem?
323  if (!(MouseConfig.flags & FLG_INIT_DONE)) {
324  ginput_lld_mouse_init();
325 
326  #if GINPUT_MOUSE_NEED_CALIBRATION
327  #if GINPUT_MOUSE_LLD_CALIBRATION_LOADSAVE
328  if (!MouseConfig.fnloadcal) {
329  MouseConfig.fnloadcal = ginput_lld_mouse_calibration_load;
330  MouseConfig.flags &= ~FLG_CAL_FREE;
331  }
332  if (!MouseConfig.fnsavecal)
333  MouseConfig.fnsavecal = ginput_lld_mouse_calibration_save;
334  #endif
335  if (MouseConfig.fnloadcal && (pc = (Calibration *)MouseConfig.fnloadcal(instance))) {
336  MouseConfig.caldata = pc[0];
337  MouseConfig.flags |= (FLG_CAL_OK|FLG_CAL_SAVED);
338  if ((MouseConfig.flags & FLG_CAL_FREE))
339  gfxFree((void *)pc);
340  } else if (instance == 9999) {
341  MouseConfig.caldata.ax = 1;
342  MouseConfig.caldata.bx = 0;
343  MouseConfig.caldata.cx = 0;
344  MouseConfig.caldata.ay = 0;
345  MouseConfig.caldata.by = 1;
346  MouseConfig.caldata.cy = 0;
347  MouseConfig.flags |= (FLG_CAL_OK|FLG_CAL_SAVED);
348  } else
349  ginputCalibrateMouse(instance);
350  #endif
351 
352  // Get the first reading
353  MouseConfig.last_buttons = 0;
354  get_calibrated_reading(&MouseConfig.t);
355 
356  // Mark init as done and start the Poll timer
357  MouseConfig.flags |= FLG_INIT_DONE;
358  gtimerStart(&MouseTimer, MousePoll, 0, TRUE, GINPUT_MOUSE_POLL_PERIOD);
359  }
360 
361  // Return our structure as the handle
362  return (GSourceHandle)&MouseConfig;
363 }

365 bool_t ginputGetMouseStatus(uint16_t instance, GEventMouse *pe) {
366  // Win32 threads don't seem to recognise priority and/or pre-emption
367  // so we add a sleep here to prevent 100% polled applications from locking up.
368  gfxSleepMilliseconds(1);
369 
370  if (instance || (MouseConfig.flags & (FLG_INIT_DONE|FLG_IN_CAL)) != FLG_INIT_DONE)
371  return FALSE;
372 
373  pe->type = GINPUT_MOUSE_EVENT_TYPE;
374  pe->instance = instance;
375  pe->x = MouseConfig.t.x;
376  pe->y = MouseConfig.t.y;
377  pe->z = MouseConfig.t.z;
378  pe->current_buttons = MouseConfig.t.buttons;
379  pe->last_buttons = MouseConfig.last_buttons;
380  if (pe->current_buttons & ~pe->last_buttons & GINPUT_MOUSE_BTN_LEFT)
381  pe->meta = GMETA_MOUSE_DOWN;
382  else if (~pe->current_buttons & pe->last_buttons & GINPUT_MOUSE_BTN_LEFT)
383  pe->meta = GMETA_MOUSE_UP;
384  else
385  pe->meta = GMETA_NONE;
386  return TRUE;
387 }

389 bool_t ginputCalibrateMouse(uint16_t instance) {
390  #if !GINPUT_MOUSE_NEED_CALIBRATION
391  (void) instance;
392 
393  return FALSE;
394  #else
395 
396  const coord_t height = gdispGetHeight();
397  const coord_t width = gdispGetWidth();
398  const MousePoint cross[] = {{(width / 4), (height / 4)},
399  {(width - (width / 4)) , (height / 4)},
400  {(width - (width / 4)) , (height - (height / 4))},
401  {(width / 2), (height / 2)}}; /* Check point */
402  MousePoint points[GINPUT_MOUSE_CALIBRATION_POINTS];
403  const MousePoint *pc;
404  MousePoint *pt;
405  int32_t px, py;
406  unsigned i, j;
407  font_t font1, font2;
408  #if GINPUT_MOUSE_MAX_CALIBRATION_ERROR >= 0
409  unsigned err;
410  #endif
411 
412  if (instance || (MouseConfig.flags & FLG_IN_CAL))
413  return FALSE;
414 
415  font1 = gdispOpenFont(GINPUT_MOUSE_CALIBRATION_FONT);
416  font2 = gdispOpenFont(GINPUT_MOUSE_CALIBRATION_FONT2);
417 
418  MouseConfig.flags |= FLG_IN_CAL;
419  gtimerStop(&MouseTimer);
420  MouseConfig.flags &= ~(FLG_CAL_OK|FLG_CAL_SAVED);
421 
422  #if GDISP_NEED_CONTROL
423  gdispSetOrientation(GDISP_ROTATE_0);
424  #endif
425 
426  #if GDISP_NEED_CLIP
427  gdispSetClip(0, 0, width, height);
428  #endif
429 
430  #if GINPUT_MOUSE_MAX_CALIBRATION_ERROR >= 0
431  while(1) {
432  #endif
433  gdispClear(Blue);
434 
435  gdispFillStringBox(0, 5, width, 30, GINPUT_MOUSE_CALIBRATION_TEXT, font1, White, Blue, justifyCenter);
436 
437  for(i = 0, pt = points, pc = cross; i < GINPUT_MOUSE_CALIBRATION_POINTS; i++, pt++, pc++) {
438  _tsDrawCross(pc);
439 
440  do {
441 
442  /* Wait for the mouse to be pressed */
443  while(get_raw_reading(&MouseConfig.t), !(MouseConfig.t.buttons & GINPUT_MOUSE_BTN_LEFT))
444  gfxSleepMilliseconds(20);
445 
446  /* Average all the samples while the mouse is down */
447  for(px = py = 0, j = 0;
448  gfxSleepMilliseconds(20), /* Settling time between readings */
449  get_raw_reading(&MouseConfig.t),
450  (MouseConfig.t.buttons & GINPUT_MOUSE_BTN_LEFT);
451  j++) {
452  px += MouseConfig.t.x;
453  py += MouseConfig.t.y;
454  }
455 
456  } while(!j);
457 
458  pt->x = px / j;
459  pt->y = py / j;
460 
461  _tsClearCross(pc);
462 
463  if (i >= 1 && pt->x == (pt-1)->x && pt->y == (pt-1)->y) {
464  gdispFillStringBox(0, 35, width, 40, GINPUT_MOUSE_CALIBRATION_SAME_TEXT, font2, Red, Yellow, justifyCenter);
465  gfxSleepMilliseconds(5000);
466  gdispFillArea(0, 35, width, 40, Blue);
467  }
468 
469  }
470 
471  /* Apply 3 point calibration algorithm */
472  _tsDo3PointCalibration(cross, points, &MouseConfig.caldata);
473 
474  /* Verification of correctness of calibration (optional) :
475  * See if the 4th point (Middle of the screen) coincides with the calibrated
476  * result. If point is within +/- Squareroot(ERROR) pixel margin, then successful calibration
477  * Else, start from the beginning.
478  */
479  #if GINPUT_MOUSE_MAX_CALIBRATION_ERROR >= 0
480  /* Transform the co-ordinates */
481  MouseConfig.t.x = points[3].x;
482  MouseConfig.t.y = points[3].y;
483  _tsTransform(&MouseConfig.t, &MouseConfig.caldata);
484 
485  /* Calculate the delta */
486  err = (MouseConfig.t.x - cross[3].x) * (MouseConfig.t.x - cross[3].x) +
487  (MouseConfig.t.y - cross[3].y) * (MouseConfig.t.y - cross[3].y);
488 
489  if (err <= GINPUT_MOUSE_MAX_CALIBRATION_ERROR * GINPUT_MOUSE_MAX_CALIBRATION_ERROR)
490  break;
491 
492  gdispFillStringBox(0, 35, width, 40, GINPUT_MOUSE_CALIBRATION_ERROR_TEXT, font2, Red, Yellow, justifyCenter);
493  gfxSleepMilliseconds(5000);
494  }
495  #endif
496 
497  // Restart everything
498  gdispCloseFont(font1);
499  gdispCloseFont(font2);
500  MouseConfig.flags |= FLG_CAL_OK;
501  MouseConfig.last_buttons = 0;
502  get_calibrated_reading(&MouseConfig.t);
503  MouseConfig.flags &= ~FLG_IN_CAL;
504  if ((MouseConfig.flags & FLG_INIT_DONE))
505  gtimerStart(&MouseTimer, MousePoll, 0, TRUE, GINPUT_MOUSE_POLL_PERIOD);
506 
507  // Save the calibration data (if possible)
508  if (MouseConfig.fnsavecal) {
509  MouseConfig.fnsavecal(instance, (const uint8_t *)&MouseConfig.caldata, sizeof(MouseConfig.caldata));
510  MouseConfig.flags |= FLG_CAL_SAVED;
511  }
512 
513  // Clear the screen using the GWIN default background color
514  #if GFX_USE_GWIN
515  gdispClear(gwinGetDefaultBgColor());
516  #else
517  gdispClear(Black);
518  #endif
519 
520  return TRUE;
521  #endif
522 }

524 /* Set the routines to save and fetch calibration data.
525  * This function should be called before first calling ginputGetMouse() for a particular instance
526  * as the gdispGetMouse() routine may attempt to fetch calibration data and perform a startup calibration if there is no way to get it.
527  * If this is called after gdispGetMouse() has been called and the driver requires calibration storage, it will immediately save the data is has already obtained.
528  * The 'requireFree' parameter indicates if the fetch buffer must be free()'d to deallocate the buffer provided by the Fetch routine.
529  */
530 void ginputSetMouseCalibrationRoutines(uint16_t instance, GMouseCalibrationSaveRoutine fnsave, GMouseCalibrationLoadRoutine fnload, bool_t requireFree) {
531  #if GINPUT_MOUSE_NEED_CALIBRATION
532  if (instance)
533  return;
534 
535  MouseConfig.fnloadcal = fnload;
536  MouseConfig.fnsavecal = fnsave;
537  if (requireFree)
538  MouseConfig.flags |= FLG_CAL_FREE;
539  else
540  MouseConfig.flags &= ~FLG_CAL_FREE;
