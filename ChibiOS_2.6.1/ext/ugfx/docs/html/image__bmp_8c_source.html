77 #if GUARANTEED_LITTLE_ENDIAN

81 #else

121 gdispImageError gdispImageOpen_BMP(gdispImage *img) {
122  gdispImagePrivate *priv;
123  uint8_t hdr[2];
124  uint16_t aword;
125  uint32_t adword;
126  uint32_t offsetColorTable;
127 
128  /* Read the file identifier */
129  if (img->io.fns->read(&img->io, hdr, 2) != 2)
130  return GDISP_IMAGE_ERR_BADFORMAT; // It can't be us
131 
132  /* Process the BITMAPFILEHEADER structure */
133 
134  /**
135  * We only accept Windows V2+ bitmaps.
136  * - we don't support OS/2 bitmaps, icons, pointers, or Windows V1 bitmaps.
137  */
138  if (hdr[0] != 'B' || hdr[1] != 'M')
139  return GDISP_IMAGE_ERR_BADFORMAT; // It can't be us
140 
141  /* We know we are a BMP format image */
142  img->flags = 0;
143 
144  /* Allocate our private area */
145  if (!(img->priv = (gdispImagePrivate *)gdispImageAlloc(img, sizeof(gdispImagePrivate))))
146  return GDISP_IMAGE_ERR_NOMEMORY;
147 
148  /* Initialise the essential bits in the private area */
149  priv = img->priv;
150  priv->frame0cache = 0;
151  priv->bmpflags = 0;
152 #if GDISP_NEED_IMAGE_BMP_1 || GDISP_NEED_IMAGE_BMP_4 || GDISP_NEED_IMAGE_BMP_4_RLE || GDISP_NEED_IMAGE_BMP_8 || GDISP_NEED_IMAGE_BMP_8_RLE
153  priv->palette = 0;
154 #endif
155 
156  /* Skip the size field and the 2 reserved fields */
157  if (img->io.fns->read(&img->io, priv->buf, 8) != 8)
158  goto baddatacleanup;
159 
160  /* Get the offset to the bitmap data */
161  if (img->io.fns->read(&img->io, &priv->frame0pos, 4) != 4)
162  goto baddatacleanup;
163  CONVERT_FROM_DWORD_LE(priv->frame0pos);
164 
165  /* Process the BITMAPCOREHEADER structure */
166 
167  /* Get the offset to the colour data */
168  if (img->io.fns->read(&img->io, &offsetColorTable, 4) != 4)
169  goto baddatacleanup;
170  CONVERT_FROM_DWORD_LE(offsetColorTable);
171  offsetColorTable += 14; // Add the size of the BITMAPFILEHEADER
172 
173  // Detect our bitmap version
174  if (offsetColorTable == 12+14) {
175  img->priv->bmpflags |= BMP_V2;
176 
177  // Read the header
178  if (img->io.fns->read(&img->io, priv->buf, 12-4) != 12-4)
179  goto baddatacleanup;
180  // Get the width
181  img->width = *(uint16_t *)(((uint8_t *)priv->buf)+0);
182  CONVERT_FROM_WORD_LE(img->width);
183  // Get the height
184  img->height = *(uint16_t *)(((uint8_t *)priv->buf)+2);
185  CONVERT_FROM_WORD_LE(img->height);
186  if (img->height < 0) {
187  img->priv->bmpflags |= BMP_TOP_TO_BOTTOM;
188  img->height = -img->height;
189  }
190  // Get the planes
191  aword = *(uint16_t *)(((uint8_t *)priv->buf)+4);
192  CONVERT_FROM_WORD_LE(aword);
193  if (aword != 1)
194  goto unsupportedcleanup;
195  // Get the bits per pixel
196  aword = *(uint16_t *)(((uint8_t *)priv->buf)+6);
197  CONVERT_FROM_WORD_LE(aword);
198  switch(aword) {
199 #if GDISP_NEED_IMAGE_BMP_1
200  case 1:
201 #endif
202 #if GDISP_NEED_IMAGE_BMP_4
203  case 4:
204 #endif
205 #if GDISP_NEED_IMAGE_BMP_8
206  case 8:
207 #endif
208 #if GDISP_NEED_IMAGE_BMP_1 || GDISP_NEED_IMAGE_BMP_4 || GDISP_NEED_IMAGE_BMP_8
209  priv->bmpflags |= BMP_PALETTE;
210  priv->palsize = 1<<aword;
211  break;
212 #endif
213 #if GDISP_NEED_IMAGE_BMP_24
214  case 24:
215  break;
216 #endif
217  default:
218  goto unsupportedcleanup;
219  }
220  priv->bitsperpixel = aword;
221 
222  } else if (offsetColorTable >= 40+14) {
223  if (offsetColorTable > 40+14)
224  priv->bmpflags |= BMP_V4;
225 
226  // Read the header
227  if (img->io.fns->read(&img->io, priv->buf, 40-4) != 40-4)
228  goto baddatacleanup;
229  // Get the width
230  adword = *(uint32_t *)(((uint8_t *)priv->buf)+0);
231  CONVERT_FROM_DWORD_LE(adword);
232  if (adword > 32768) // This also picks up negative values
233  goto unsupportedcleanup;
234  img->width = adword;
235  // Get the height
236  adword = *(uint32_t *)(((uint8_t *)priv->buf)+4);
237  CONVERT_FROM_DWORD_LE(adword);
238  if ((int32_t)adword < 0) { // Negative test
239  priv->bmpflags |= BMP_TOP_TO_BOTTOM;
240  adword = -adword;
241  }
242  if (adword > 32768)
243  goto unsupportedcleanup;
244  img->height = adword;
245  // Get the planes
246  aword = *(uint16_t *)(((uint8_t *)priv->buf)+8);
247  CONVERT_FROM_WORD_LE(aword);
248  if (aword != 1)
249  goto unsupportedcleanup;
250  // Get the bits per pixel
251  aword = *(uint16_t *)(((uint8_t *)priv->buf)+10);
252  CONVERT_FROM_WORD_LE(aword);
253  switch(aword) {
254 #if GDISP_NEED_IMAGE_BMP_1
255  case 1:
256 #endif
257 #if GDISP_NEED_IMAGE_BMP_4 || GDISP_NEED_IMAGE_BMP_4_RLE
258  case 4:
259 #endif
260 #if GDISP_NEED_IMAGE_BMP_8 || GDISP_NEED_IMAGE_BMP_8_RLE
261  case 8:
262 #endif
263 #if GDISP_NEED_IMAGE_BMP_1 || GDISP_NEED_IMAGE_BMP_4 || GDISP_NEED_IMAGE_BMP_4_RLE || GDISP_NEED_IMAGE_BMP_8 || GDISP_NEED_IMAGE_BMP_8_RLE
264  priv->bmpflags |= BMP_PALETTE;
265  priv->palsize = 1<<aword;
266  break;
267 #endif
268 #if GDISP_NEED_IMAGE_BMP_16
269  case 16:
270 #endif
271 #if GDISP_NEED_IMAGE_BMP_24
272  case 24:
273 #endif
274 #if GDISP_NEED_IMAGE_BMP_32
275  case 32:
276 #endif
277 #if GDISP_NEED_IMAGE_BMP_16 || GDISP_NEED_IMAGE_BMP_24 || GDISP_NEED_IMAGE_BMP_32
278  break;
279 #endif
280  default:
281  goto unsupportedcleanup;
282  }
283  priv->bitsperpixel = aword;
284  // Get the compression
285  adword = *(uint32_t *)(((uint8_t *)priv->buf)+12);
286  CONVERT_FROM_DWORD_LE(adword);
287  switch(adword) {
288  case 0: // BI_RGB - uncompressed
289  break;
290 #if GDISP_NEED_IMAGE_BMP_8_RLE
291  case 1: // BI_RLE8 compression
292  if (priv->bitsperpixel != 8)
293  goto unsupportedcleanup;
294  priv->bmpflags |= BMP_COMP_RLE;
295  break;
296 #endif
297 #if GDISP_NEED_IMAGE_BMP_4_RLE
298  case 2: // BI_RLE4 compression
299  if (priv->bitsperpixel != 4)
300  goto unsupportedcleanup;
301  priv->bmpflags |= BMP_COMP_RLE;
302  break;
303 #endif
304 #if GDISP_NEED_IMAGE_BMP_16 || GDISP_NEED_IMAGE_BMP_32
305  case 3: // BI_BITFIELDS decoding
306  if (priv->bitsperpixel < 16 || priv->bitsperpixel == 24)
307  goto unsupportedcleanup;
308  priv->bmpflags |= BMP_COMP_MASK;
309  if (priv->bmpflags & BMP_V4) // V4 stored the masks in the header
310  offsetColorTable = 40+14;
311  break;
312 #endif
313  default:
314  goto unsupportedcleanup;
315  }
316  priv->bitsperpixel = aword;
317 #if GDISP_NEED_IMAGE_BMP_1 || GDISP_NEED_IMAGE_BMP_4 || GDISP_NEED_IMAGE_BMP_4_RLE || GDISP_NEED_IMAGE_BMP_8 || GDISP_NEED_IMAGE_BMP_8_RLE
318  // Get the actual colors used
319  adword = *(uint32_t *)(((uint8_t *)priv->buf)+28);
320  CONVERT_FROM_DWORD_LE(adword);
321  if (adword && adword < priv->palsize)
322  priv->palsize = adword;
323 #endif
324  } else
325  goto baddatacleanup;
326 
327 #if GDISP_NEED_IMAGE_BMP_1 || GDISP_NEED_IMAGE_BMP_4 || GDISP_NEED_IMAGE_BMP_4_RLE || GDISP_NEED_IMAGE_BMP_8 || GDISP_NEED_IMAGE_BMP_8_RLE
328  /* Load the palette tables */
329  if (priv->bmpflags & BMP_PALETTE) {
330  img->io.fns->seek(&img->io, offsetColorTable);
331 
332  if (!(priv->palette = (color_t *)gdispImageAlloc(img, priv->palsize*sizeof(color_t))))
333  return GDISP_IMAGE_ERR_NOMEMORY;
334  if (priv->bmpflags & BMP_V2) {
335  for(aword = 0; aword < priv->palsize; aword++) {
336  if (img->io.fns->read(&img->io, &priv->buf, 3) != 3) goto baddatacleanup;
337  priv->palette[aword] = RGB2COLOR(((uint8_t *)priv->buf)[2], ((uint8_t *)priv->buf)[1], ((uint8_t *)priv->buf)[0]);
338  }
339  } else {
340  for(aword = 0; aword < priv->palsize; aword++) {
341  if (img->io.fns->read(&img->io, &priv->buf, 4) != 4) goto baddatacleanup;
342  priv->palette[aword] = RGB2COLOR(((uint8_t *)priv->buf)[2], ((uint8_t *)priv->buf)[1], ((uint8_t *)priv->buf)[0]);
343  }
344  }
345 
346  }
347 #endif
348 
349 #if GDISP_NEED_IMAGE_BMP_16 || GDISP_NEED_IMAGE_BMP_32
350  /* Load the bit masks */
351  if (priv->bmpflags & BMP_COMP_MASK) {
352  img->io.fns->seek(&img->io, offsetColorTable);
353  if (img->io.fns->read(&img->io, &priv->maskred, 4) != 4) goto baddatacleanup;
354  CONVERT_FROM_DWORD_LE(priv->maskred);
355  if (img->io.fns->read(&img->io, &priv->maskgreen, 4) != 4) goto baddatacleanup;
356  CONVERT_FROM_DWORD_LE(priv->maskgreen);
357  if (img->io.fns->read(&img->io, &priv->maskblue, 4) != 4) goto baddatacleanup;
358  CONVERT_FROM_DWORD_LE(priv->maskblue);
359  if (priv->bmpflags & BMP_V4) {
360  if (img->io.fns->read(&img->io, &priv->maskalpha, 4) != 4) goto baddatacleanup;
361  CONVERT_FROM_DWORD_LE(priv->maskalpha);
362  } else
363  priv->maskalpha = 0;
364  } else if (priv->bitsperpixel == 16) {
365  priv->bmpflags |= BMP_COMP_MASK;
366  priv->maskred = 0x7C00;
367  priv->maskgreen = 0x03E0;
368  priv->maskblue = 0x001F;
369  priv->maskalpha = 0;
370  } else if (priv->bitsperpixel == 32) {
371  priv->bmpflags |= BMP_COMP_MASK;
372  priv->maskred = 0x00FF0000;
373  priv->maskgreen = 0x0000FF00;
374  priv->maskblue = 0x000000FF;
375  priv->maskalpha = 0;
376  }
377 
378  /* We need to adjust the masks and calculate the shift values so the result scales 0 -> 255 */
379  if (priv->bmpflags & BMP_COMP_MASK) {
380  priv->shiftred = 0;
381  priv->shiftgreen = 0;
382  priv->shiftblue = 0;
383  if (priv->maskred) {
384  if (priv->maskred < 256)
385  for(adword = priv->maskred; adword < 128; priv->shiftred--, adword <<= 1);
386  else
387  for(adword = priv->maskred; adword > 255; priv->shiftred++, adword >>= 1);
388  }
389  if (priv->maskgreen) {
390  if (priv->maskgreen < 256)
391  for(adword = priv->maskgreen; adword < 128; priv->shiftgreen--, adword <<= 1);
392  else
393  for(adword = priv->maskgreen; adword > 255; priv->shiftgreen++, adword >>= 1);
394  }
395  if (priv->maskblue) {
396  if (priv->maskblue < 256)
397  for(adword = priv->maskblue; adword < 128; priv->shiftblue--, adword <<= 1);
398  else
399  for(adword = priv->maskblue; adword > 255; priv->shiftblue++, adword >>= 1);
400  }
401  if (priv->maskalpha) {
402  if (priv->maskalpha < 256)
403  for(adword = priv->maskalpha; adword < 128; priv->shiftalpha--, adword <<= 1);
404  else
405  for(adword = priv->maskalpha; adword > 255; priv->shiftalpha++, adword >>= 1);
406  }
407  }
408 #endif
409 
410  img->type = GDISP_IMAGE_TYPE_BMP;
411  return GDISP_IMAGE_ERR_OK;
412

420 }

436 static coord_t getPixels(gdispImage *img, coord_t x) {
437  gdispImagePrivate * priv;
438  color_t * pc;
439  coord_t len;
440 
441  priv = img->priv;
442  pc = priv->buf;
443  len = 0;
444 
445  switch(priv->bitsperpixel) {
446 #if GDISP_NEED_IMAGE_BMP_1
447  case 1:
448  {
449  uint8_t b[4];
450  uint8_t m;
451 
452  priv = img->priv;
453  pc = priv->buf;
454  len = 0;
455 
456  while(x < img->width && len <= BLIT_BUFFER_SIZE-32) {
457  if (img->io.fns->read(&img->io, &b, 4) != 4)
458  return 0;
459 
460  for(m=0x80; m; m >>= 1, pc++)
461  pc[0] = priv->palette[(m&b[0]) ? 1 : 0];
462  for(m=0x80; m; m >>= 1, pc++)
463  pc[0] = priv->palette[(m&b[1]) ? 1 : 0];
464  for(m=0x80; m; m >>= 1, pc++)
465  pc[0] = priv->palette[(m&b[2]) ? 1 : 0];
466  for(m=0x80; m; m >>= 1, pc++)
467  pc[0] = priv->palette[(m&b[3]) ? 1 : 0];
468  len += 32;
469  x += 32;
470  }
471  }
472  return len;
473 #endif
474 
475 #if GDISP_NEED_IMAGE_BMP_4 || GDISP_NEED_IMAGE_BMP_4_RLE
476  case 4:
477  #if GDISP_NEED_IMAGE_BMP_4_RLE
478  #if GDISP_NEED_IMAGE_BMP_4
479  if (priv->bmpflags & BMP_COMP_RLE)
480  #endif
481  {
482  uint8_t b[4];
483 
484  while(x < img->width) {
485  if (priv->bmpflags & BMP_RLE_ENC) {
486  while (priv->rlerun && len <= BLIT_BUFFER_SIZE-2 && x < img->width) {
487  *pc++ = priv->palette[priv->rlecode >> 4];
488  priv->rlerun--;
489  len++;
490  x++;
491  if (priv->rlerun) {
492  *pc++ = priv->palette[priv->rlecode & 0x0F];
493  priv->rlerun--;
494  len++;
495  x++;
496  }
497  }
498  if (priv->rlerun) // Return if we have more run to do
499  return len;
500  } else if (priv->bmpflags & BMP_RLE_ABS) {
501  while (priv->rlerun && len <= BLIT_BUFFER_SIZE-2 && x < img->width) {
502  if (img->io.fns->read(&img->io, &b, 1) != 1)
503  return 0;
504  *pc++ = priv->palette[b[0] >> 4];
505  priv->rlerun--;
506  len++;
507  x++;
508  if (priv->rlerun) {
509  *pc++ = priv->palette[b[0] & 0x0F];
510  priv->rlerun--;
511  len++;
512  x++;
513  }
514  }
515  if (priv->rlerun) // Return if we have more run to do
516  return len;
517  if ((img->io.pos - priv->frame0pos)&1) { // Make sure we are on a word boundary
518  if (img->io.fns->read(&img->io, &b, 1) != 1)
519  return 0;
520  }
521  }
522 
523  // We have finished the current run - read a new run
524  priv->bmpflags &= ~(BMP_RLE_ENC|BMP_RLE_ABS);
525 
526  // There are always at least 2 bytes in an RLE code
527  if (img->io.fns->read(&img->io, &b, 2) != 2)
528  return 0;
529 
530  if (b[0]) { // Encoded mode
531  priv->rlerun = b[0];
532  priv->rlecode = b[1];
533  priv->bmpflags |= BMP_RLE_ENC;
534  } else if (b[1] == 0) { // End of line
535  if (x < img->width) {
536  priv->rlerun = img->width - x;
537  priv->rlecode = 0; // Who knows what color this should really be
538  priv->bmpflags |= BMP_RLE_ENC;
539  }
540  } else if (b[1] == 1) { // End of file
541  return len;

776  gdispImagePrivate * priv;
777  color_t * pcs;
778  color_t * pcd;
779  coord_t pos, x, y;
780  size_t len;
781 
782  /* If we are already cached - just return OK */
783  priv = img->priv;
784  if (priv->frame0cache)
785  return GDISP_IMAGE_ERR_OK;
786 
787  /* We need to allocate the cache */
788  len = img->width * img->height * sizeof(pixel_t);
789  priv->frame0cache = (pixel_t *)gdispImageAlloc(img, len);
790  if (!priv->frame0cache)
791  return GDISP_IMAGE_ERR_NOMEMORY;
792

799

804  if (!pos) {

808  }
809  *pcd++ = *pcs++;
810  x++; pos--;
811  }
812  }
813  } else {

815  x = 0; pos = 0;
