47 static void gwinLabelDefaultDraw(GWidgetObject *gw, void *param) {
48  coord_t w, h;
49  (void) param;
50 
51  w = (gw->g.flags & GLABEL_FLG_WAUTO) ? getwidth(gw->text, gw->g.font, gdispGetWidth() - gw->g.x) : gw->g.width;
52  h = (gw->g.flags & GLABEL_FLG_HAUTO) ? getheight(gw->text, gw->g.font, gdispGetWidth() - gw->g.x) : gw->g.height;
53 
54  if (gw->g.width != w || gw->g.height != h) {
55  gwinResize(&gw->g, w, h);
56 
57  return;
58  }
59 
60  // render the text
61  gdispFillStringBox(gw->g.x, gw->g.y, gw->g.width, gw->g.height, gw->text, gw->g.font,
62  (gw->g.flags & GWIN_FLG_ENABLED) ? gw->pstyle->enabled.text : gw->pstyle->disabled.text, gw->pstyle->background,
63  justifyLeft);
64 
65  // render the border (if any)
66  if (gw->g.flags & GLABEL_FLG_BORDER)
67  gdispDrawBox(gw->g.x, gw->g.y, gw->g.width, gw->g.height, (gw->g.flags & GWIN_FLG_ENABLED) ? gw->pstyle->enabled.edge : gw->pstyle->disabled.edge);
68 }

105 GHandle gwinLabelCreate(GLabelObject *widget, GWidgetInit *pInit) {
106  uint16_t flags = 0;
107 
108  // auto assign width
109  if (pInit->g.width <= 0) {
110 
111  flags |= GLABEL_FLG_WAUTO;
112  pInit->g.width = getwidth(pInit->text, gwinGetDefaultFont(), gdispGetWidth() - pInit->g.x);
113  }
114 
115  // auto assign height
116  if (pInit->g.height <= 0) {
117  flags |= GLABEL_FLG_HAUTO;
118  pInit->g.height = getheight(pInit->text, gwinGetDefaultFont(), gdispGetWidth() - pInit->g.x);
119  }
120 
121  if (!(widget = (GLabelObject *)_gwidgetCreate(&widget->w, pInit, &labelVMT)))
122  return 0;
123 
124  // no borders by default
125  flags &=~ GLABEL_FLG_BORDER;
126 
127  widget->w.g.flags |= flags;
128  gwinSetVisible(&widget->w.g, pInit->g.show);
129 
130  return (GHandle)widget;
131 }
